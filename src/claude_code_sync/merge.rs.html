6/// Represents a node in the conversation message tree.
7///
8/// Each node contains a conversation entry and can have multiple children,
9/// allowing for branching conversations where a single message has multiple
10/// different continuations (e.g., when edited on different machines).
11#[derive(Debug, Clone)]
12pub struct MessageNode {
13    /// The conversation entry for this node
14    pub entry: ConversationEntry,
15
16    /// Child nodes (messages that have this message as their parent)
17    pub children: Vec<MessageNode>,
18}
19
20impl MessageNode {
21    /// Creates a new message node with no children
22    fn new(entry: ConversationEntry) -> Self {
23        MessageNode {
24            entry,
25            children: Vec::new(),
26        }
27    }
28
29    /// Adds a child node to this message
30    fn add_child(&mut self, child: MessageNode) {
31        self.children.push(child);
32    }
33
34    /// Recursively collects all entries in this subtree in depth-first order
35    fn collect_entries(&self) -> Vec<ConversationEntry> {
36        let mut entries = vec![self.entry.clone()];
37
38        // Sort children by timestamp to maintain chronological order
39        let mut sorted_children = self.children.clone();
40        sorted_children.sort_by(|a, b| {
41            let a_ts = a.entry.timestamp.as_ref();
42            let b_ts = b.entry.timestamp.as_ref();
43            a_ts.cmp(&b_ts)
44        });
45
46        for child in &sorted_children {
47            entries.extend(child.collect_entries());
48        }
49
50        entries
51    }
52}

97    /// Creates a new smart merger for the given sessions
98    pub fn new(local: &'a ConversationSession, remote: &'a ConversationSession) -> Self {
99        SmartMerger {
100            local,
101            remote,
102            stats: MergeStats::default(),
103        }
104    }

106    /// Performs the smart merge and returns the result
107    pub fn merge(&mut self) -> Result<MergeResult> {
108        // Count initial messages
109        self.stats.local_messages = self.local.message_count();
110        self.stats.remote_messages = self.remote.message_count();
111
112        // Build UUID maps for both sessions
113        let local_map = self.build_uuid_map(&self.local.entries);
114        let remote_map = self.build_uuid_map(&self.remote.entries);
115
116        // Detect and resolve edits (same UUID, different content)
117        let resolved_edits = self.detect_and_resolve_edits(&local_map, &remote_map)?;
118
119        // Separate entries into UUID-tracked and non-UUID entries
120        let (local_uuid_entries, local_non_uuid): (Vec<_>, Vec<_>) =
121            self.local.entries.iter().partition(|e| e.uuid.is_some());
122
123        let (remote_uuid_entries, remote_non_uuid): (Vec<_>, Vec<_>) =
124            self.remote.entries.iter().partition(|e| e.uuid.is_some());
125
126        // Combine all UUID entries from both sides
127        let mut all_uuid_entries: Vec<&ConversationEntry> = Vec::new();
128        all_uuid_entries.extend(local_uuid_entries);
129        all_uuid_entries.extend(remote_uuid_entries);
130
131        // Build a single unified tree from all entries
132        let merged_roots = self.build_unified_tree(&all_uuid_entries, &resolved_edits)?;
133
134        // Flatten tree back to entries
135        let mut merged_entries = Vec::new();
136        for root in &merged_roots {
137            merged_entries.extend(root.collect_entries());
138        }
139
140        // Merge non-UUID entries by timestamp
141        let local_vec: Vec<_> = local_non_uuid.into_iter().cloned().collect();
142        let remote_vec: Vec<_> = remote_non_uuid.into_iter().cloned().collect();
143        let non_uuid_merged = self.merge_by_timestamp(&local_vec, &remote_vec);
144
145        self.stats.timestamp_merged = non_uuid_merged.len();
146
147        // Combine UUID-based and timestamp-based entries, sorted by timestamp
148        merged_entries.extend(non_uuid_merged);
149        merged_entries.sort_by(|a, b| {
150            let a_ts = a.timestamp.as_ref();
151            let b_ts = b.timestamp.as_ref();
152            a_ts.cmp(&b_ts)
153        });
154
155        self.stats.merged_messages = merged_entries.len();
156
157        Ok(MergeResult {
158            merged_entries,
159            stats: self.stats.clone(),
160        })
161    }

163    /// Builds a UUID to entry map
164    fn build_uuid_map(&self, entries: &[ConversationEntry]) -> HashMap<String, ConversationEntry> {
165        entries
166            .iter()
167            .filter_map(|e| e.uuid.as_ref().map(|uuid| (uuid.clone(), e.clone())))
168            .collect()

171    /// Detects edits (same UUID, different content) and resolves them by timestamp
172    fn detect_and_resolve_edits(
173        &mut self,
174        local_map: &HashMap<String, ConversationEntry>,
175        remote_map: &HashMap<String, ConversationEntry>,
176    ) -> Result<HashMap<String, ConversationEntry>> {
177        let mut resolved = HashMap::new();
178
179        // Find all UUIDs that exist in both maps
180        let common_uuids: HashSet<_> = local_map
181            .keys()
182            .filter(|uuid| remote_map.contains_key(*uuid))
183            .collect();
184
185        for uuid in common_uuids {
186            let local_entry = &local_map[uuid];
187            let remote_entry = &remote_map[uuid];
188
189            // Compare content to detect edits
190            let local_json = serde_json::to_string(local_entry)?;
191            let remote_json = serde_json::to_string(remote_entry)?;
192
193            if local_json != remote_json {
194                // Edit detected - resolve by timestamp
195                self.stats.edits_resolved += 1;
196
197                let chosen = self.resolve_by_timestamp(local_entry, remote_entry);
198                resolved.insert(uuid.clone(), chosen.clone());
199            } else {
200                // Same content, just add one copy
201                resolved.insert(uuid.clone(), local_entry.clone());
202            }
203        }
204
205        Ok(resolved)
206    }

208    /// Resolves an edit conflict by choosing the entry with the newer timestamp
209    fn resolve_by_timestamp<'b>(
210        &self,
211        local: &'b ConversationEntry,
212        remote: &'b ConversationEntry,
213    ) -> &'b ConversationEntry {
214        match (&local.timestamp, &remote.timestamp) {
215            (Some(local_ts), Some(remote_ts)) => {
216                if remote_ts > local_ts {
217                    remote
218                } else {
219                    local
220                }
221            }
222            (Some(_), None) => local,
223            (None, Some(_)) => remote,
224            (None, None) => local, // Fallback to local if no timestamps
225        }
226    }

228    /// Builds a unified tree from all entries (both local and remote)
229    fn build_unified_tree(
230        &mut self,
231        all_entries: &[&ConversationEntry],
232        resolved_edits: &HashMap<String, ConversationEntry>,
233    ) -> Result<Vec<MessageNode>> {
234        // Build UUID map, preferring resolved edits and deduplicating
235        let mut uuid_to_entry: HashMap<String, ConversationEntry> = HashMap::new();
236
237        for entry in all_entries {
238            if let Some(uuid) = &entry.uuid {
239                // Only insert if not already present (avoids duplicates)
240                if !uuid_to_entry.contains_key(uuid) {
241                    // Use resolved edit if available, otherwise use original entry
242                    let entry_to_use = resolved_edits.get(uuid).unwrap_or(*entry);
243                    uuid_to_entry.insert(uuid.clone(), entry_to_use.clone());
244                }
245            }
246        }
247
248        // Map parent UUID -> list of child UUIDs
249        let mut parent_to_children: HashMap<Option<String>, Vec<String>> = HashMap::new();
250
251        for (uuid, entry) in &uuid_to_entry {
252            parent_to_children
253                .entry(entry.parent_uuid.clone())
254                .or_default()
255                .push(uuid.clone());
256        }
257
258        // Build tree recursively
259        fn build_subtree(
260            uuid: &str,
261            uuid_to_entry: &HashMap<String, ConversationEntry>,
262            parent_to_children: &HashMap<Option<String>, Vec<String>>,
263        ) -> MessageNode {
264            let entry = uuid_to_entry.get(uuid).unwrap().clone();
265            let mut node = MessageNode::new(entry);
266
267            // Get children for this UUID
268            if let Some(child_uuids) = parent_to_children.get(&Some(uuid.to_string())) {
269                for child_uuid in child_uuids {
270                    let child_node = build_subtree(child_uuid, uuid_to_entry, parent_to_children);
271                    node.add_child(child_node);
272                }
273            }
274
275            node
276        }
277
278        // Find root UUIDs (entries with no parent)
279        let root_uuids = parent_to_children.get(&None).cloned().unwrap_or_default();
280
281        // Build trees from each root
282        let mut roots = Vec::new();
283        for root_uuid in root_uuids {
284            let root_node = build_subtree(&root_uuid, &uuid_to_entry, &parent_to_children);
285            roots.push(root_node);
286        }
287
288        // Count branches
289        for entry in uuid_to_entry.values() {
290            if let Some(children) = parent_to_children.get(&Some(entry.uuid.clone().unwrap())) {
291                if children.len() > 1 {
292                    self.stats.branches_detected += 1;
293                }
294            }
295        }
296
297        // Sort roots by timestamp
298        roots.sort_by(|a, b| {
299            let a_ts = a.entry.timestamp.as_ref();
300            let b_ts = b.entry.timestamp.as_ref();
301            a_ts.cmp(&b_ts)
302        });
303
304        Ok(roots)
305    }

307    /// Builds a message tree from entries
308    #[allow(dead_code)]
309    fn build_tree(
310        &mut self,
311        entries: &[&ConversationEntry],
312        resolved_edits: &HashMap<String, ConversationEntry>,
313    ) -> Result<Vec<MessageNode>> {
314        // Build UUID map, preferring resolved edits
315        let mut uuid_to_entry: HashMap<String, ConversationEntry> = HashMap::new();
316
317        for entry in entries {
318            if let Some(uuid) = &entry.uuid {
319                // Use resolved edit if available, otherwise use original entry
320                let entry_to_use = resolved_edits.get(uuid).unwrap_or(*entry);
321                uuid_to_entry.insert(uuid.clone(), entry_to_use.clone());
322            }
323        }
324
325        // Track which UUIDs have been used as children
326        let mut used_as_child = HashSet::new();
327
328        // Map parent UUID -> list of child nodes
329        let mut parent_to_children: HashMap<Option<String>, Vec<MessageNode>> = HashMap::new();
330
331        for entry in uuid_to_entry.values() {
332            let node = MessageNode::new(entry.clone());
333
334            parent_to_children
335                .entry(entry.parent_uuid.clone())
336                .or_default()
337                .push(node);
338
339            if let Some(parent_uuid) = &entry.parent_uuid {
340                used_as_child.insert(parent_uuid.clone());
341            }
342        }
343
344        // Build tree recursively
345        fn build_subtree(
346            parent_uuid: Option<String>,
347            parent_to_children: &HashMap<Option<String>, Vec<MessageNode>>,
348        ) -> Vec<MessageNode> {
349            if let Some(children) = parent_to_children.get(&parent_uuid) {
350                children
351                    .iter()
352                    .map(|child| {
353                        let mut node = child.clone();
354                        let child_uuid = child.entry.uuid.clone();
355                        node.children = build_subtree(child_uuid, parent_to_children);
356                        node
357                    })
358                    .collect()
359            } else {
360                Vec::new()
361            }
362        }
363
364        // Find root nodes (entries with no parent or parent not in this tree)
365        let roots = build_subtree(None, &parent_to_children);
366
367        // Also find orphaned subtrees (entries whose parent exists but isn't in our tree)
368        let mut all_roots = roots;
369        for (parent_uuid, children) in parent_to_children.iter() {
370            if let Some(parent) = parent_uuid {
371                if !uuid_to_entry.contains_key(parent) {
372                    all_roots.extend(children.iter().map(|child| {
373                        let mut node = child.clone();
374                        let child_uuid = child.entry.uuid.clone();
375                        node.children = build_subtree(child_uuid, &parent_to_children);
376                        node
377                    }));
378                }
379            }
380        }
381
382        Ok(all_roots)
383    }

385    /// Merges two message trees, keeping all branches
386    #[allow(dead_code)]
387    fn merge_trees(
388        &mut self,
389        local_roots: Vec<MessageNode>,
390        remote_roots: Vec<MessageNode>,
391    ) -> Result<Vec<MessageNode>> {
392        // Create a map of UUID -> MessageNode for efficient lookup
393        let mut merged_nodes: HashMap<String, MessageNode> = HashMap::new();
394
395        // Process all nodes from both trees (recursively collect all nodes, not just roots)
396        fn collect_nodes_recursive(
397            nodes: &[MessageNode],
398            collected: &mut HashMap<String, MessageNode>,
399        ) {
400            for node in nodes {
401                if let Some(uuid) = &node.entry.uuid {
402                    collected.insert(uuid.clone(), node.clone());
403                }
404                // Recursively collect children
405                collect_nodes_recursive(&node.children, collected);
406            }
407        }
408
409        collect_nodes_recursive(&local_roots, &mut merged_nodes);
410
411        // Merge remote nodes, combining children where nodes have same UUID
412        for remote_root in remote_roots {
413            self.merge_node_into(&remote_root, &mut merged_nodes);
414        }
415
416        // Detect branches (nodes with multiple children)
417        self.count_branches(&merged_nodes);
418
419        // Extract root nodes
420        let mut roots: Vec<MessageNode> = merged_nodes
421            .values()
422            .filter(|node| node.entry.parent_uuid.is_none())
423            .cloned()
424            .collect();
425
426        // Sort roots by timestamp
427        roots.sort_by(|a, b| {
428            let a_ts = a.entry.timestamp.as_ref();
429            let b_ts = b.entry.timestamp.as_ref();
430            a_ts.cmp(&b_ts)
431        });
432
433        Ok(roots)
434    }

436    /// Merges a node into the existing tree
437    #[allow(dead_code)]
438    #[allow(clippy::only_used_in_recursion)]
439    fn merge_node_into(
440        &mut self,
441        node: &MessageNode,
442        merged_nodes: &mut HashMap<String, MessageNode>,
443    ) {
444        if let Some(uuid) = &node.entry.uuid {
445            // Collect children that need to be merged recursively
446            let mut children_to_merge = Vec::new();
447
448            if let Some(existing) = merged_nodes.get_mut(uuid) {
449                // Node already exists, merge children
450                for child in &node.children {
451                    // Check if this child already exists
452                    let child_exists = if let Some(child_uuid) = &child.entry.uuid {
453                        existing
454                            .children
455                            .iter()
456                            .any(|c| c.entry.uuid.as_ref() == Some(child_uuid))
457                    } else {
458                        false
459                    };
460
461                    if !child_exists {
462                        // Add new child
463                        existing.add_child(child.clone());
464                    }
465
466                    // Collect child for recursive merge
467                    if child.entry.uuid.is_some() {
468                        children_to_merge.push(child.clone());
469                    }
470                }
471            } else {
472                // New node, add it
473                merged_nodes.insert(uuid.clone(), node.clone());
474
475                // Collect all children for recursive merge
476                for child in &node.children {
477                    children_to_merge.push(child.clone());
478                }
479            }
480
481            // Now recursively merge all collected children
482            for child in &children_to_merge {
483                self.merge_node_into(child, merged_nodes);
484            }
485        }
486    }

488    /// Counts the number of branches in the tree
489    #[allow(dead_code)]
490    fn count_branches(&mut self, nodes: &HashMap<String, MessageNode>) {
491        for node in nodes.values() {
492            if node.children.len() > 1 {
493                self.stats.branches_detected += 1;
494            }
495        }
496    }

498    /// Merges non-UUID entries by timestamp, removing duplicates
499    fn merge_by_timestamp(
500        &mut self,
501        local: &[ConversationEntry],
502        remote: &[ConversationEntry],
503    ) -> Vec<ConversationEntry> {
504        let mut all_entries = local.to_owned();
505        all_entries.extend(remote.to_owned());
506
507        // Sort by timestamp
508        all_entries.sort_by(|a, b| {
509            let a_ts = a.timestamp.as_ref();
510            let b_ts = b.timestamp.as_ref();
511            a_ts.cmp(&b_ts)
512        });
513
514        // Remove duplicates by comparing JSON representation
515        let mut seen = HashSet::new();
516        let mut unique_entries = Vec::new();
517
518        for entry in all_entries {
519            if let Ok(json) = serde_json::to_string(&entry) {
520                if seen.insert(json.clone()) {
521                    unique_entries.push(entry);
522                } else {
523                    self.stats.duplicates_removed += 1;
524                }
525            }
526        }
527
528        unique_entries
529    }

532/// Attempts to perform a smart merge on two conversation sessions
533///
534/// This is the main entry point for the smart merge feature. It will attempt
535/// to intelligently combine messages from both conversations, handling:
536/// - Non-overlapping messages (simple merge)
537/// - Edited messages (resolved by timestamp)
538/// - Conversation branches (all branches preserved)
539/// - Entries without UUIDs (merged by timestamp)
540///
541/// # Arguments
542///
543/// * `local` - The local conversation session
544/// * `remote` - The remote conversation session
545///
546/// # Returns
547///
548/// Returns `Ok(MergeResult)` if merge succeeds, or an error if the merge
549/// cannot be completed (e.g., due to corrupted data or circular references).
550pub fn merge_conversations(
551    local: &ConversationSession,
552    remote: &ConversationSession,
553) -> Result<MergeResult> {
554    // Validate sessions have same session ID
555    if local.session_id != remote.session_id {
556        return Err(anyhow!(
557            "Cannot merge conversations with different session IDs: {} vs {}",
558            local.session_id,
559            remote.session_id
560        ));
561    }
562
563    let mut merger = SmartMerger::new(local, remote);
564    merger.merge()
565}
