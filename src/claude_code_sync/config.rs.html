4/// Cross-platform configuration directory manager
5pub struct ConfigManager;
6
7impl ConfigManager {
8    /// Get the main configuration directory path following platform conventions:
9    /// - Linux: $XDG_CONFIG_HOME/claude-code-sync or ~/.config/claude-code-sync
10    /// - macOS: ~/Library/Application Support/claude-code-sync
11    /// - Windows: %APPDATA%\claude-code-sync
12    ///
13    /// The `CLAUDE_CODE_SYNC_CONFIG_DIR` environment variable overrides the
14    /// platform default on every OS. This is primarily what keeps the test suite
15    /// isolated from a real user's config: on macOS `dirs`/the platform default
16    /// ignores `XDG_CONFIG_HOME`, so without an explicit, cross-platform override a
17    /// `cargo test` run would read and clobber the user's real config directory.
18    pub fn config_dir() -> Result<PathBuf> {
19        // Explicit override, honored on all platforms, checked before any default.
20        if let Ok(override_dir) = std::env::var("CLAUDE_CODE_SYNC_CONFIG_DIR") {
21            if !override_dir.is_empty() {
22                return Ok(PathBuf::from(override_dir).join("claude-code-sync"));
23            }
24        }
25
26        #[cfg(target_os = "linux")]
27        {
28            // Follow XDG Base Directory Specification
29            if let Ok(xdg_config) = std::env::var("XDG_CONFIG_HOME") {
30                Ok(PathBuf::from(xdg_config).join("claude-code-sync"))
31            } else {
32                let home = dirs::home_dir().context("Failed to get home directory")?;
33                Ok(home.join(".config").join("claude-code-sync"))
34            }
35        }
36
37        #[cfg(target_os = "macos")]
38        {
39            // Follow macOS conventions
40            let home = dirs::home_dir().context("Failed to get home directory")?;
41            Ok(home
42                .join("Library")
43                .join("Application Support")
44                .join("claude-code-sync"))
45        }
46
47        #[cfg(target_os = "windows")]
48        {
49            // Use Windows APPDATA
50            Ok(dirs::config_dir()
51                .context("Failed to get Windows config directory")?
52                .join("claude-code-sync"))
53        }
54
55        #[cfg(not(any(target_os = "linux", target_os = "macos", target_os = "windows")))]
56        {
57            // Fallback for other platforms
58            let home = dirs::home_dir().context("Failed to get home directory")?;
59            Ok(home.join(".claude-code-sync"))
60        }
61    }

99    /// Ensure the configuration directory exists
100    pub fn ensure_config_dir() -> Result<PathBuf> {
101        let config_dir = Self::config_dir()?;
102        std::fs::create_dir_all(&config_dir).with_context(|| {
103            format!(
104                "Failed to create config directory: {}",
105                config_dir.display()
106            )
107        })?;
108        Ok(config_dir)
109    }
110
111    /// Ensure the snapshots directory exists
112    #[allow(dead_code)]
113    pub fn ensure_snapshots_dir() -> Result<PathBuf> {
114        let snapshots_dir = Self::snapshots_dir()?;

118                snapshots_dir.display()
119            )
120        })?;
121        Ok(snapshots_dir)
122    }
