1use anyhow::{Context, Result};

4/// Cross-platform configuration directory manager
5pub struct ConfigManager;
6
7impl ConfigManager {
8    /// Get the main configuration directory path following platform conventions:
9    /// - Linux: $XDG_CONFIG_HOME/claude-code-sync or ~/.config/claude-code-sync
10    /// - macOS: ~/Library/Application Support/claude-code-sync
11    /// - Windows: %APPDATA%\claude-code-sync
12    pub fn config_dir() -> Result<PathBuf> {
13        #[cfg(target_os = "linux")]
14        {
15            // Follow XDG Base Directory Specification
16            if let Ok(xdg_config) = std::env::var("XDG_CONFIG_HOME") {
17                Ok(PathBuf::from(xdg_config).join("claude-code-sync"))
18            } else {
19                let home = dirs::home_dir().context("Failed to get home directory")?;
20                Ok(home.join(".config").join("claude-code-sync"))
21            }
22        }
23
24        #[cfg(target_os = "macos")]
25        {
26            // Follow macOS conventions
27            let home = dirs::home_dir().context("Failed to get home directory")?;
28            Ok(home
29                .join("Library")
30                .join("Application Support")
31                .join("claude-code-sync"))
32        }
33
34        #[cfg(target_os = "windows")]
35        {
36            // Use Windows APPDATA
37            Ok(dirs::config_dir()
38                .context("Failed to get Windows config directory")?
39                .join("claude-code-sync"))
40        }
41
42        #[cfg(not(any(target_os = "linux", target_os = "macos", target_os = "windows")))]
43        {
44            // Fallback for other platforms
45            let home = dirs::home_dir().context("Failed to get home directory")?;
46            Ok(home.join(".claude-code-sync"))
47        }
48    }

79    }
80
81    /// Get the log file path
82    pub fn log_file_path() -> Result<PathBuf> {
83        Ok(Self::config_dir()?.join("claude-code-sync.log"))
84    }
85
86    /// Ensure the configuration directory exists
87    pub fn ensure_config_dir() -> Result<PathBuf> {
88        let config_dir = Self::config_dir()?;
89        std::fs::create_dir_all(&config_dir).with_context(|| {
90            format!(
91                "Failed to create config directory: {}",
92                config_dir.display()
93            )
94        })?;
95        Ok(config_dir)
96    }
97
98    /// Ensure the snapshots directory exists
99    #[allow(dead_code)]
100    pub fn ensure_snapshots_dir() -> Result<PathBuf> {
101        let snapshots_dir = Self::snapshots_dir()?;
102        std::fs::create_dir_all(&snapshots_dir).with_context(|| {
103            format!(
104                "Failed to create snapshots directory: {}",
105                snapshots_dir.display()
106            )
107        })?;
108        Ok(snapshots_dir)
109    }
