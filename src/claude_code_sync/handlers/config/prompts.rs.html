1//! Terminal interaction shared by the interactive and wizard config modes.
2//!
3//! Separate from `fields.rs` because everything here writes to stdout or opens
4//! an `inquire` prompt. Keeping the two apart is what lets `fields.rs` be tested
5//! without a TTY.

15/// Display a compact configuration summary.
16pub(super) fn display_config_summary(config: &FilterConfig) {
17    println!(
18        "  {} {}",
19        "Exclude older than:".cyan(),
20        config
21            .exclude_older_than_days
22            .map(|d| format!("{d} days"))
23            .unwrap_or_else(|| "Not set".dimmed().to_string())
24    );
25
26    println!(
27        "  {} {}",
28        "Include patterns:".cyan(),
29        format_patterns(
30            &config.include_patterns,
31            &"None (all included)".dimmed().to_string()
32        )
33    );
34
35    println!(
36        "  {} {}",
37        "Exclude patterns:".cyan(),
38        format_patterns(&config.exclude_patterns, &"None".dimmed().to_string())
39    );
40
41    println!(
42        "  {} {} MB",
43        "Max file size:".cyan(),
44        format_size_mb(config.max_file_size_bytes)
45    );
46
47    println!(
48        "  {} {}",
49        "Exclude attachments:".cyan(),
50        if config.exclude_attachments {
51            "Yes (only .jsonl files)".green().to_string()
52        } else {
53            "No (all files)".yellow().to_string()
54        }
55    );
56
57    let enabled: Vec<&str> = toggleable()
58        .filter(|d| config.sync_artifacts.is_enabled(d.id))
59        .map(|d| d.name)
60        .collect();
61    println!(
62        "  {} {}",
63        "Artifact sync:".cyan(),
64        if enabled.is_empty() {
65            "All disabled".dimmed().to_string()
66        } else {
67            enabled.join(", ")
68        }
69    );
70}

72/// The current age filter, rendered for a "Current: ..." line.
73pub(super) fn current_age(config: &FilterConfig) -> String {
74    format_age_days(config.exclude_older_than_days)

77/// MultiSelect over all artifact categories, pre-selecting the currently
78/// enabled ones. Returns the resulting toggles.
79pub(super) fn prompt_artifact_toggle_selection(
80    current: &ArtifactToggles,
81) -> Result<ArtifactToggles> {
82    let rows: Vec<_> = toggleable().collect();
83    let options: Vec<String> = rows
84        .iter()
85        .map(|d| format!("{} — {}", d.name, d.description))
86        .collect();
87    let preselected: Vec<usize> = rows
88        .iter()
89        .enumerate()
90        .filter(|(_, d)| current.is_enabled(d.id))
91        .map(|(i, _)| i)
92        .collect();
93
94    let picked = MultiSelect::new("Artifact categories to sync:", options)
95        .with_default(&preselected)
96        .with_help_message(
97            "Space toggles, Enter confirms. Secrets (credentials, settings.local.json, \
98             .env*, keys) are never synced regardless of selection.",
99        )
100        .prompt()
101        .context("Failed to get artifact category selection")?;
102
103    let mut toggles = ArtifactToggles::default();
104    for label in picked {
105        let name = label.split(" — ").next().unwrap_or(&label);
106        if let Some(desc) = find_by_name(name) {
107            toggles.set_enabled(desc.id, true);
108        }
109    }
110    Ok(toggles)
111}
