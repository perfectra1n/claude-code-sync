13/// Try to recover an existing repo if state.json is missing but repo exists
14///
15/// This handles the case where a user has a valid repo in the default location
16/// but the state.json file is missing (e.g., from an older version or deletion).
17fn try_recover_existing_repo() -> Result<Option<MultiRepoState>> {
18    let default_repo = match ConfigManager::default_repo_dir() {
19        Ok(path) => path,
20        Err(_) => return Ok(None),
21    };
22
23    if !default_repo.exists() || !scm::is_repo(&default_repo) {
24        return Ok(None);
25    }
26
27    let (has_remote, remote_url) = match scm::open(&default_repo) {
28        Ok(repo) => {
29            let has_remote = repo.has_remote("origin");
30            let remote_url = if has_remote {
31                repo.get_remote_url("origin").ok()
32            } else {
33                None
34            };
35            (has_remote, remote_url)
36        }
37        Err(_) => (false, None),
38    };
39
40    println!(
41        "{} Found existing repo at: {}",
42        "!".yellow(),
43        default_repo.display()
44    );
45    if let Some(ref url) = remote_url {
46        println!("  Remote: {}", url.cyan());
47    }
48    println!("  Recovering configuration...");
49    println!();
50
51    let repo_config = RepoConfig {
52        name: "default".to_string(),
53        sync_repo_path: default_repo,
54        has_remote,
55        is_cloned_repo: false, // We can't know this for sure
56        remote_url,
57        description: Some("Recovered from existing repository".to_string()),
58    };
59
60    let mut repos = HashMap::new();
61    repos.insert("default".to_string(), repo_config);
62
63    let state = MultiRepoState {
64        version: 2,
65        active_repo: "default".to_string(),
66        repos,
67    };
68
69    state.save()?;
70
71    Ok(Some(state))
72}

74/// Handle the repository selector menu
75///
76/// Shows when `claude-code-sync config` is run with no arguments.
77/// Displays all configured repositories and allows switching between them.
78pub fn handle_repo_selector() -> Result<()> {
79    println!("{}", "Repository Configuration".cyan().bold());
80    println!("{}", "=".repeat(60).cyan());
81    println!();
82
83    // "Not initialized" is recoverable if a repo happens to sit in the default
84    // location; any other load failure is not ours to interpret.
85    let mut state = match MultiRepoState::load() {
86        Ok(s) => s,
87        Err(e) => {
88            let err_msg = e.to_string();
89            if err_msg.contains("not initialized")
90                || err_msg.contains("Run 'claude-code-sync init'")
91            {
92                if let Some(recovered) = try_recover_existing_repo()? {
93                    println!(
94                        "{}",
95                        "Found existing repository - recovered configuration!".green()
96                    );
97                    println!();
98                    recovered
99                } else {
100                    println!("{}", "No repositories configured.".yellow());
101                    println!();
102                    println!(
103                        "Run '{}' to set up your first repository.",
104                        "claude-code-sync init".cyan()
105                    );
106                    return Ok(());
107                }
108            } else {
109                return Err(e);
110            }
111        }
112    };
113
114    if state.repos.is_empty() {
115        println!("{}", "No repositories configured.".yellow());
116        println!();
117        println!(
118            "Run '{}' to set up your first repository.",
119            "claude-code-sync init".cyan()
120        );
121        return Ok(());
122    }
123
124    // Active repo first, then alphabetical.
125    let mut repo_entries: Vec<_> = state.repos.values().collect();
126    repo_entries.sort_by(|a, b| {
127        if a.name == state.active_repo {
128            std::cmp::Ordering::Less
129        } else if b.name == state.active_repo {
130            std::cmp::Ordering::Greater
131        } else {
132            a.name.cmp(&b.name)
133        }
134    });
135
136    let mut options: Vec<String> = repo_entries
137        .iter()
138        .map(|repo| {
139            let active_marker = if repo.name == state.active_repo {
140                format!(" {}", "[ACTIVE]".green().bold())
141            } else {
142                String::new()
143            };
144
145            let path_str = repo.sync_repo_path.display().to_string();
146            let remote_info = repo
147                .remote_url
148                .as_ref()
149                .map(|u| format!(" ({})", u.dimmed()))
150                .unwrap_or_default();
151
152            format!(
153                "{}{} - {}{}",
154                repo.name, active_marker, path_str, remote_info
155            )
156        })
157        .collect();
158
159    options.push(format!("{}", "─── Actions ───".dimmed()));
160    options.push("Configure filters (current repo)".to_string());
161    options.push("Exit".to_string());
162
163    let selection = Select::new("Select a repository to make active:", options.clone())
164        .with_help_message("Use arrow keys to navigate, Enter to select")
165        .prompt()
166        .context("Failed to get user selection")?;
167
168    if selection.contains("─── Actions ───") || selection == "Exit" {
169        return Ok(());
170    }
171
172    if selection == "Configure filters (current repo)" {
173        return handle_config_interactive();
174    }
175
176    // The repo name is the first token, before any [ACTIVE] marker.
177    let repo_name = selection
178        .split_whitespace()
179        .next()
180        .ok_or_else(|| anyhow::anyhow!("Invalid selection"))?;
181
182    if repo_name == state.active_repo {
183        println!();
184        println!(
185            "{} '{}' is already the active repository.",
186            "ℹ".blue(),
187            repo_name.cyan()
188        );
189        return Ok(());
190    }
191
192    if state.repos.contains_key(repo_name) {
193        state.active_repo = repo_name.to_string();
194        state.save()?;
195
196        println!();
197        println!(
198            "{} Switched to repository '{}'",
199            "✓".green().bold(),
200            repo_name.cyan()
201        );
202
203        if let Some(repo) = state.repos.get(repo_name) {
204            println!("  Path: {}", repo.sync_repo_path.display());
205            if let Some(ref url) = repo.remote_url {
206                println!("  Remote: {url}");
207            }
208            if repo.has_remote {
209                println!("  Has remote: {}", "Yes".green());
210            } else {
211                println!("  Has remote: {}", "No (local only)".yellow());
212            }
213        }
214    } else {
215        return Err(anyhow::anyhow!("Repository '{}' not found", repo_name));
216    }
217
218    Ok(())
219}
