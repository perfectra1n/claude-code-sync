1//! Pure value logic for the config commands.
2//!
3//! Nothing in here prompts, prints, or touches the filesystem — which is the
4//! point. These conversions used to be inlined into the `inquire` call sites in
5//! `interactive.rs` and `wizard.rs`, once per mode, which meant they could only
6//! be exercised with a TTY attached and so were never tested at all.
7
8use anyhow::{bail, Context, Result};
9
10const BYTES_PER_MB: f64 = 1024.0 * 1024.0;
11
12/// Split a comma-separated pattern list into its trimmed, non-empty parts.
13///
14/// This is deliberately only the split/trim/filter core. What an *empty* input
15/// means is left to the caller, because the two modes report it differently:
16/// interactive says "Cleared include patterns", the wizard says "Include
17/// patterns set: []".
18pub(super) fn parse_patterns(input: &str) -> Vec<String> {
19    input
20        .split(',')
21        .map(|s| s.trim().to_string())
22        .filter(|s| !s.is_empty())
23        .collect()
24}
25
26/// Parse a file-size limit expressed in megabytes.
27///
28/// Rejects non-positive and non-finite values. Rust's float-to-int cast
29/// saturates rather than wrapping or erroring, so without this check `-5` would
30/// quietly become a 0-byte limit — which makes `FilterConfig::should_include`
31/// reject every file — and `1e30` would become `u64::MAX`.
32pub(super) fn parse_file_size_mb(input: &str) -> Result<u64> {
33    let mb: f64 = input
34        .trim()
35        .parse()
36        .context("Invalid number. Must be a positive number.")?;
37
38    if !mb.is_finite() {
39        bail!("Max file size must be a finite number of megabytes");
40    }
41    if mb <= 0.0 {
42        bail!("Max file size must be greater than 0 MB (got {mb})");
43    }
44
45    Ok((mb * BYTES_PER_MB) as u64)
46}
47
48/// Render a byte count as megabytes with one decimal place.
49pub(super) fn format_size_mb(bytes: u64) -> String {
50    format!("{:.1}", bytes as f64 / BYTES_PER_MB)
51}
52
53/// The "current value" text for the optional age filter.
54pub(super) fn format_age_days(days: Option<u32>) -> String {
55    days.map_or_else(|| "Not set".to_string(), |d| d.to_string())
56}
57
58/// The "current value" text for a pattern list, with a caller-chosen label for
59/// the empty case (the two modes word it differently).
60pub(super) fn format_patterns(patterns: &[String], when_empty: &str) -> String {
61    if patterns.is_empty() {
62        when_empty.to_string()
63    } else {
64        patterns.join(", ")
65    }
66}

68#[cfg(test)]
69mod tests {
70    use super::*;
71
72    #[test]
73    fn test_parse_patterns_splits_and_trims() {
74        assert_eq!(
75            parse_patterns("*work*, /home/user/x , *test*"),
76            vec!["*work*", "/home/user/x", "*test*"]
77        );
78    }
79
80    #[test]
81    fn test_parse_patterns_drops_empty_segments() {
82        // Trailing commas and stray whitespace are what people actually type.
83        assert_eq!(parse_patterns("a,,b,"), vec!["a", "b"]);
84        assert_eq!(parse_patterns("  ,  ,  "), Vec::<String>::new());
85    }
86
87    #[test]
88    fn test_parse_patterns_empty_input_yields_empty_list() {
89        assert_eq!(parse_patterns(""), Vec::<String>::new());
90    }
91
92    #[test]
93    fn test_parse_file_size_accepts_positive_values() {
94        assert_eq!(parse_file_size_mb("10").unwrap(), 10 * 1024 * 1024);
95        assert_eq!(parse_file_size_mb("0.5").unwrap(), 512 * 1024);
96        assert_eq!(parse_file_size_mb("  2  ").unwrap(), 2 * 1024 * 1024);
97    }
98
99    #[test]
100    fn test_parse_file_size_rejects_negative() {
101        // The bug this function exists to prevent: `as u64` saturates, so a
102        // negative input used to land as a 0-byte limit, silently filtering out
103        // every single file instead of erroring.
104        let err = parse_file_size_mb("-5").unwrap_err().to_string();
105        assert!(err.contains("greater than 0"), "unexpected error: {err}");
106    }
107
108    #[test]
109    fn test_parse_file_size_rejects_zero() {
110        assert!(parse_file_size_mb("0").is_err());
111    }
112
113    #[test]
114    fn test_parse_file_size_rejects_non_finite() {
115        // `"inf".parse::<f64>()` succeeds, and `inf as u64` saturates to u64::MAX.
116        assert!(parse_file_size_mb("inf").is_err());
117        assert!(parse_file_size_mb("NaN").is_err());
118    }
119
120    #[test]
121    fn test_parse_file_size_rejects_garbage() {
122        assert!(parse_file_size_mb("ten").is_err());
123        assert!(parse_file_size_mb("").is_err());
124    }
125
126    #[test]
127    fn test_format_size_mb_round_trips_with_parse() {
128        let bytes = parse_file_size_mb("12.5").unwrap();
129        assert_eq!(format_size_mb(bytes), "12.5");
130    }
131
132    #[test]
133    fn test_format_age_days() {
134        assert_eq!(format_age_days(Some(30)), "30");
135        assert_eq!(format_age_days(None), "Not set");
136    }
137
138    #[test]
139    fn test_format_patterns_uses_caller_label_when_empty() {
140        assert_eq!(format_patterns(&[], "None"), "None");
141        assert_eq!(
142            format_patterns(&["a".to_string(), "b".to_string()], "None"),
143            "a, b"
144        );
145    }
146}
