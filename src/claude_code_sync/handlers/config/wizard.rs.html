1//! Wizard-mode configuration: walk every setting in order, gating each one
2//! behind a yes/no question.
3//!
4//! This is intentionally not the same UX as `interactive.rs`. The wizard asks
5//! "Do you want to ...?" first and only prompts for a value on yes, and it has
6//! no "empty clears the setting" affordance — an empty pattern list here simply
7//! parses to `[]` and is reported as such. Both flows share the value logic in
8//! `fields.rs`, not the prompt shape.
9
10use anyhow::{Context, Result};
11use colored::Colorize;
12use inquire::{Confirm, Text};
13
14use super::fields::{format_patterns, format_size_mb, parse_file_size_mb, parse_patterns};
15use super::prompts::{current_age, display_config_summary, prompt_artifact_toggle_selection};
16use crate::filter::FilterConfig;
17
18/// Handle wizard-mode configuration
19///
20/// Steps through each configuration option one by one
21pub fn handle_config_wizard() -> Result<()> {
22    println!("{}", "Configuration Wizard".cyan().bold());
23    println!("{}", "=".repeat(80).cyan());
24    println!();
25    println!(
26        "{}",
27        "This wizard will walk you through all configuration options.".dimmed()
28    );
29    println!(
30        "{}",
31        "Press Enter to keep current value or enter a new value.".dimmed()
32    );
33    println!();
34
35    let current_config = FilterConfig::load().context("Failed to load current configuration")?;
36    let mut modified_config = current_config.clone();
37
38    // 1. Exclude older than
39    println!("{}", "1. Age Filter".bold().cyan());
40    println!("   Current: {}", current_age(&modified_config).yellow());
41
42    let exclude_old =
43        Confirm::new("Do you want to exclude projects older than a certain number of days?")
44            .with_default(modified_config.exclude_older_than_days.is_some())
45            .prompt()?;
46
47    if exclude_old {
48        let default_days = modified_config
49            .exclude_older_than_days
50            .unwrap_or(30)
51            .to_string();
52        let input = Text::new("How many days?")
53            .with_default(&default_days)
54            .prompt()?;
55
56        let days: u32 = input
57            .trim()
58            .parse()
59            .context("Invalid number. Must be a positive integer.")?;
60        modified_config.exclude_older_than_days = Some(days);
61        println!(
62            "  {} Will exclude projects older than {} days\n",
63            "✓".green(),
64            days
65        );
66    } else {
67        modified_config.exclude_older_than_days = None;
68        println!("  {} Age filter disabled\n", "✓".green());
69    }
70
71    // 2. Include patterns
72    println!("{}", "2. Include Patterns".bold().cyan());
73    println!(
74        "   Current: {}",
75        format_patterns(
76            &modified_config.include_patterns,
77            "None (all projects included)"
78        )
79        .yellow()
80    );
81
82    let use_include = Confirm::new("Do you want to limit sync to specific project patterns?")
83        .with_default(!modified_config.include_patterns.is_empty())
84        .with_help_message("Example: *work*, /home/user/important/*")
85        .prompt()?;
86
87    if use_include {
88        let default = modified_config.include_patterns.join(", ");
89        let input = Text::new("Enter include patterns (comma-separated):")
90            .with_default(&default)
91            .with_help_message("Glob patterns like '*work*' or '/specific/path'")
92            .prompt()?;
93
94        modified_config.include_patterns = parse_patterns(&input);
95        println!(
96            "  {} Include patterns set: {:?}\n",
97            "✓".green(),
98            modified_config.include_patterns
99        );
100    } else {
101        modified_config.include_patterns = Vec::new();
102        println!("  {} All projects will be included\n", "✓".green());
103    }
104
105    // 3. Exclude patterns
106    println!("{}", "3. Exclude Patterns".bold().cyan());
107    println!(
108        "   Current: {}",
109        format_patterns(&modified_config.exclude_patterns, "None").yellow()
110    );
111
112    let use_exclude = Confirm::new("Do you want to exclude specific project patterns?")
113        .with_default(!modified_config.exclude_patterns.is_empty())
114        .with_help_message("Example: *test*, *tmp*, /temp/*")
115        .prompt()?;
116
117    if use_exclude {
118        let default = modified_config.exclude_patterns.join(", ");
119        let input = Text::new("Enter exclude patterns (comma-separated):")
120            .with_default(&default)
121            .with_help_message("Glob patterns like '*test*' or '/tmp/*'")
122            .prompt()?;
123
124        modified_config.exclude_patterns = parse_patterns(&input);
125        println!(
126            "  {} Exclude patterns set: {:?}\n",
127            "✓".green(),
128            modified_config.exclude_patterns
129        );
130    } else {
131        modified_config.exclude_patterns = Vec::new();
132        println!("  {} No exclusion patterns\n", "✓".green());
133    }
134
135    // 4. Exclude attachments
136    println!("{}", "4. File Type Filter".bold().cyan());
137    println!(
138        "   Current: {}",
139        if modified_config.exclude_attachments {
140            "Exclude attachments".yellow()
141        } else {
142            "Include all files".yellow()
143        }
144    );
145
146    let exclude_attachments = Confirm::new("Exclude attachments (images, PDFs, etc.)?")
147        .with_default(modified_config.exclude_attachments)
148        .with_help_message("If yes, only .jsonl conversation files will be synced")
149        .prompt()?;
150
151    modified_config.exclude_attachments = exclude_attachments;
152    println!(
153        "  {} Attachments will be {}\n",
154        "✓".green(),
155        if exclude_attachments {
156            "excluded"
157        } else {
158            "included"
159        }
160    );
161
162    // 5. Max file size
163    println!("{}", "5. File Size Limit".bold().cyan());
164    println!(
165        "   Current: {} MB",
166        format_size_mb(modified_config.max_file_size_bytes)
167    );
168
169    let change_size = Confirm::new("Do you want to change the maximum file size limit?")
170        .with_default(false)
171        .prompt()?;
172
173    if change_size {
174        let input = Text::new("Max file size (MB):")
175            .with_default(&format_size_mb(modified_config.max_file_size_bytes))
176            .prompt()?;
177
178        modified_config.max_file_size_bytes = parse_file_size_mb(&input)?;
179        println!(
180            "  {} Max file size set to {} MB\n",
181            "✓".green(),
182            format_size_mb(modified_config.max_file_size_bytes)
183        );
184    } else {
185        println!("  {} Keeping current max file size\n", "✓".green());
186    }
187
188    // Artifact sync categories
189    let change_artifacts =
190        Confirm::new("Configure artifact sync categories (settings, skills, agents, ...)?")
191            .with_default(false)
192            .prompt()?;
193    if change_artifacts {
194        modified_config.sync_artifacts =
195            prompt_artifact_toggle_selection(&modified_config.sync_artifacts)?;
196    }
197
198    // Summary and confirmation
199    println!("{}", "=".repeat(80).cyan());
200    println!("{}", "Configuration Summary:".bold().cyan());
201    println!("{}", "=".repeat(80).cyan());
202    display_config_summary(&modified_config);
203    println!();
204
205    let confirm = Confirm::new("Save this configuration?")
206        .with_default(true)
207        .prompt()?;
208
209    if confirm {
210        modified_config
211            .save()
212            .context("Failed to save configuration")?;
213        println!("\n{} Configuration saved successfully!", "✓".green().bold());
214    } else {
215        println!("\n{}", "Configuration not saved.".yellow());
216    }
217
218    Ok(())
219}
