8/// Represents a single line/entry in the JSONL conversation file
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct ConversationEntry {
11    /// The type of this entry (e.g., "user", "assistant", "file-history-snapshot")
12    ///
13    /// This field identifies what kind of entry this is in the conversation.
14    /// Common types include user messages, assistant responses, and system events.
15    #[serde(rename = "type")]
16    pub entry_type: String,
17
18    /// Unique identifier for this conversation entry
19    ///
20    /// Each entry may have its own UUID to uniquely identify it within the conversation.
21    /// Not all entry types require a UUID, hence this is optional.
22    #[serde(skip_serializing_if = "Option::is_none")]
23    pub uuid: Option<String>,
24
25    /// UUID of the parent entry in the conversation thread
26    ///
27    /// This links entries together in a conversation tree, allowing for branching
28    /// and threading of messages. If present, it references the UUID of the entry
29    /// that this entry is responding to or following from.
30    #[serde(rename = "parentUuid", skip_serializing_if = "Option::is_none")]
31    pub parent_uuid: Option<String>,
32
33    /// Session identifier grouping related conversation entries together
34    ///
35    /// All entries within a single conversation session share the same session ID.
36    /// This is used to associate entries across multiple files or to reconstruct
37    /// conversation context. If not present in the entry, the filename may be used.
38    #[serde(rename = "sessionId", skip_serializing_if = "Option::is_none")]
39    pub session_id: Option<String>,
40
41    /// ISO 8601 timestamp indicating when this entry was created
42    ///
43    /// Format is typically "YYYY-MM-DDTHH:MM:SS.sssZ" (e.g., "2025-01-01T00:00:00.000Z").
44    /// Used for sorting entries chronologically and determining the latest activity.
45    #[serde(skip_serializing_if = "Option::is_none")]
46    pub timestamp: Option<String>,
47
48    /// The actual message content as a JSON value
49    ///
50    /// Contains the text and structured data of the user or assistant message.
51    /// Stored as a generic JSON value to accommodate different message formats
52    /// and structures without strict schema requirements.
53    #[serde(skip_serializing_if = "Option::is_none")]
54    pub message: Option<Value>,
55
56    /// Current working directory at the time this entry was created
57    ///
58    /// Stores the filesystem path of the working directory, providing context
59    /// about where the conversation or command was executed. Useful for
60    /// reproducing environments and understanding file references.
61    #[serde(skip_serializing_if = "Option::is_none")]
62    pub cwd: Option<String>,
63
64    /// Version string of the Claude Code CLI that created this entry
65    ///
66    /// Records which version of the tool generated this conversation entry,
67    /// helpful for debugging compatibility issues and tracking feature support.
68    #[serde(skip_serializing_if = "Option::is_none")]
69    pub version: Option<String>,
70
71    /// Git branch name active when this entry was created
72    ///
73    /// Captures the current git branch context, allowing conversation entries
74    /// to be associated with specific branches in version control. Useful for
75    /// tracking which branch work was performed on.
76    #[serde(rename = "gitBranch", skip_serializing_if = "Option::is_none")]
77    pub git_branch: Option<String>,
78
79    /// Catch-all field for additional JSON properties not explicitly defined
80    ///
81    /// Preserves any extra fields in the JSON that aren't part of the explicit schema.
82    /// This allows forward compatibility - newer versions can add fields without breaking
83    /// older parsers. The flattened serde attribute merges these fields at the same level
84    /// as the named fields when serializing/deserializing.
85    #[serde(flatten)]
86    pub extra: Value,
87}

89/// Represents a complete conversation session
90#[derive(Debug, Clone)]
91pub struct ConversationSession {
92    /// Unique identifier for this conversation session
93    ///
94    /// Either extracted from the first entry that contains a sessionId field,
95    /// or derived from the filename (without extension) if no entries contain
96    /// a session ID. Used to group related conversation entries together.
97    pub session_id: String,
98
99    /// All conversation entries in chronological order
100    ///
101    /// Contains the complete sequence of entries from the JSONL file, including
102    /// user messages, assistant responses, and system events like file history
103    /// snapshots. Preserves the original order from the file.
104    pub entries: Vec<ConversationEntry>,
105
106    /// Path to the JSONL file this session was loaded from
107    ///
108    /// Stores the filesystem path of the source file, used for tracking the
109    /// origin of the conversation data and for potential file operations like
110    /// rewriting or updating the session.
111    pub file_path: String,
112}
113
114impl ConversationSession {
115    /// Parse a JSONL file into a ConversationSession
116    pub fn from_file<P: AsRef<Path>>(path: P) -> Result<Self> {
117        let path = path.as_ref();
118        let file =
119            File::open(path).with_context(|| format!("Failed to open file: {}", path.display()))?;
120
121        let reader = BufReader::new(file);
122        let mut entries = Vec::new();
123        let mut session_id = None;
124
125        for (line_num, line) in reader.lines().enumerate() {
126            let line = line.with_context(|| {
127                format!("Failed to read line {} in {}", line_num + 1, path.display())
128            })?;
129
130            if line.trim().is_empty() {
131                continue;
132            }
133
134            let entry: ConversationEntry = serde_json::from_str(&line).with_context(|| {
135                format!(
136                    "Failed to parse JSON at line {} in {}",
137                    line_num + 1,
138                    path.display()
139                )
140            })?;
141
142            // Extract session ID from first entry that has one
143            if session_id.is_none() {
144                if let Some(ref sid) = entry.session_id {
145                    session_id = Some(sid.clone());
146                }
147            }
148
149            entries.push(entry);
150        }
151
152        // If no session ID in entries, use filename (without extension) as session ID
153        let session_id = session_id
154            .or_else(|| {
155                path.file_stem()
156                    .and_then(|s| s.to_str())
157                    .map(|s| s.to_string())
158            })
159            .with_context(|| {
160                format!(
161                    "No session ID found in file or filename: {}",
162                    path.display()
163                )
164            })?;
165
166        Ok(ConversationSession {
167            session_id,
168            entries,
169            file_path: path.to_string_lossy().to_string(),
170        })
171    }
172
173    /// Write the conversation session to a JSONL file
174    pub fn write_to_file<P: AsRef<Path>>(&self, path: P) -> Result<()> {
175        let path = path.as_ref();
176
177        // Create parent directories if they don't exist
178        if let Some(parent) = path.parent() {
179            std::fs::create_dir_all(parent)
180                .with_context(|| format!("Failed to create directory: {}", parent.display()))?;
181        }
182
183        let mut file = File::create(path)
184            .with_context(|| format!("Failed to create file: {}", path.display()))?;
185
186        for entry in &self.entries {
187            let json =
188                serde_json::to_string(entry).context("Failed to serialize conversation entry")?;
189            writeln!(file, "{json}")
190                .with_context(|| format!("Failed to write to file: {}", path.display()))?;
191        }
192
193        Ok(())
194    }
195
196    /// Get the latest timestamp from the conversation
197    pub fn latest_timestamp(&self) -> Option<String> {
198        self.entries
199            .iter()
200            .filter_map(|e| e.timestamp.clone())
201            .max()
202    }
203
204    /// Get the number of messages (user + assistant) in the conversation
205    pub fn message_count(&self) -> usize {
206        self.entries
207            .iter()
208            .filter(|e| e.entry_type == "user" || e.entry_type == "assistant")
209            .count()
210    }
211
212    /// Calculate a simple hash of the conversation content
213    pub fn content_hash(&self) -> String {
214        use std::collections::hash_map::DefaultHasher;
215        use std::hash::{Hash, Hasher};
216
217        let mut hasher = DefaultHasher::new();
218        for entry in &self.entries {
219            if let Ok(json) = serde_json::to_string(entry) {
220                json.hash(&mut hasher);
221            }
222        }
223        format!("{:x}", hasher.finish())
224    }
225}

234    fn test_parse_conversation_entry() {
235        let json =
236            r#"{"type":"user","uuid":"123","sessionId":"abc","timestamp":"2025-01-01T00:00:00Z"}"#;

242    #[test]
243    fn test_read_write_session() {
244        let mut temp_file = NamedTempFile::new().unwrap();
245        writeln!(temp_file, r#"{{"type":"user","sessionId":"test-123","uuid":"1","timestamp":"2025-01-01T00:00:00Z"}}"#).unwrap();
246        writeln!(temp_file, r#"{{"type":"assistant","sessionId":"test-123","uuid":"2","timestamp":"2025-01-01T00:01:00Z"}}"#).unwrap();
247
248        let session = ConversationSession::from_file(temp_file.path()).unwrap();
249        assert_eq!(session.session_id, "test-123");
250        assert_eq!(session.entries.len(), 2);
251        assert_eq!(session.message_count(), 2);
252
253        // Test write
254        let output_temp = NamedTempFile::new().unwrap();
255        session.write_to_file(output_temp.path()).unwrap();
256
257        let reloaded = ConversationSession::from_file(output_temp.path()).unwrap();
258        assert_eq!(reloaded.session_id, session.session_id);
259        assert_eq!(reloaded.entries.len(), session.entries.len());
260    }
