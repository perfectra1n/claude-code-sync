8/// Show current remote configuration
9pub fn show_remote() -> Result<()> {
10    let state = SyncState::load()?;
11    let git_manager = GitManager::open(&state.sync_repo_path)?;
12
13    println!("{}", "=== Git Remote Configuration ===".bold().cyan());
14    println!();
15
16    // Show sync repository directory
17    println!(
18        "{} {}",
19        "Sync Directory:".bold(),
20        state.sync_repo_path.display().to_string().cyan()
21    );
22
23    // Show current branch
24    if let Ok(branch) = git_manager.current_branch() {
25        println!("{} {}", "Current Branch:".bold(), branch.cyan());
26    }
27
28    println!();
29
30    // Get repository
31    let repo = git2::Repository::open(&state.sync_repo_path)?;
32
33    // List all remotes
34    let remotes = repo.remotes()?;
35
36    if remotes.is_empty() {
37        println!("{}", "No remotes configured".yellow());
38        println!(
39            "\n{} claude-code-sync remote set origin <url>",
40            "Hint:".cyan()
41        );
42        return Ok(());
43    }
44
45    for name in remotes.iter().flatten() {
46        if let Ok(remote) = repo.find_remote(name) {
47            println!("{} {}", "Remote:".bold(), name.cyan());
48
49            if let Some(url) = remote.url() {
50                println!("  URL: {url}");
51            } else {
52                println!("  URL: {}", "None".yellow());
53            }
54
55            if let Some(push_url) = remote.pushurl() {
56                println!("  Push URL: {push_url}");
57            }
58
59            println!();
60        }
61    }
62
63    Ok(())
64}

66/// Set or update remote URL
67pub fn set_remote(name: &str, url: &str) -> Result<()> {
68    let state = SyncState::load()?;
69    let repo = git2::Repository::open(&state.sync_repo_path)?;
70
71    // Validate URL format
72    if !url.starts_with("http://") && !url.starts_with("https://") && !url.starts_with("git@") {
73        return Err(anyhow!(
74            "Invalid URL format: {url}\n\
75            \n\
76            URL must start with:\n\
77            - https:// (e.g., https://github.com/user/repo.git)\n\
78            - http:// (e.g., http://gitlab.com/user/repo.git)\n\
79            - git@ (e.g., git@github.com:user/repo.git)"
80        ));
81    }
82
83    // Check if remote exists
84    let remote_exists = repo.find_remote(name).is_ok();
85
86    if remote_exists {
87        // Update existing remote
88        repo.remote_set_url(name, url)
89            .with_context(|| format!("Failed to update remote '{name}' URL"))?;
90
91        println!(
92            "{} Updated remote '{}' to: {}",
93            "✓".green().bold(),
94            name.cyan(),
95            url
96        );
97    } else {
98        // Create new remote
99        repo.remote(name, url)
100            .with_context(|| format!("Failed to create remote '{name}'"))?;
101
102        println!(
103            "{} Created remote '{}': {}",
104            "✓".green().bold(),
105            name.cyan(),
106            url
107        );
108    }
109
110    // Update state if this is the origin remote
111    if name == "origin" {
112        let mut state = state;
113        state.has_remote = true;
114        state.save()?;
115    }
116
117    println!("\n{} claude-code-sync push", "Next:".cyan());
118
119    Ok(())
120}

122/// Remove a remote
123pub fn remove_remote(name: &str) -> Result<()> {
124    let state = SyncState::load()?;
125    let repo = git2::Repository::open(&state.sync_repo_path)?;
126
127    // Check if remote exists
128    if repo.find_remote(name).is_err() {
129        return Err(anyhow!("Remote '{name}' not found"));
130    }
131
132    // Remove the remote
133    repo.remote_delete(name)
134        .with_context(|| format!("Failed to remove remote '{name}'"))?;
135
136    println!("{} Removed remote '{}'", "✓".green().bold(), name.cyan());
137
138    // Update state if this was the origin remote
139    if name == "origin" {
140        let mut state = state;
141        state.has_remote = false;
142        state.save()?;
143    }
144
145    Ok(())
146}
