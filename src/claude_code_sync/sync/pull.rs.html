22/// Pull and merge history from sync repository
23pub fn pull_history(
24    fetch_remote: bool,
25    branch: Option<&str>,
26    interactive: bool,
27    verbosity: crate::VerbosityLevel,
28) -> Result<()> {
29    use crate::VerbosityLevel;
30
31    if verbosity != VerbosityLevel::Quiet {
32        println!("{}", "Pulling Claude Code history...".cyan().bold());
33    }
34
35    let state = SyncState::load()?;
36    let git_manager = GitManager::open(&state.sync_repo_path)?;
37    let filter = FilterConfig::load()?;
38    let claude_dir = claude_projects_dir()?;
39
40    // Get the current branch name for operation record
41    let branch_name = branch
42        .map(|s| s.to_string())
43        .or_else(|| git_manager.current_branch().ok())
44        .unwrap_or_else(|| "main".to_string());
45
46    // Fetch from remote if configured
47    if fetch_remote && state.has_remote {
48        println!("  {} from remote...", "Fetching".cyan());
49
50        match git_manager.pull("origin", &branch_name) {
51            Ok(_) => println!("  {} Pulled from origin/{}", "âœ“".green(), branch_name),
52            Err(e) => {
53                log::warn!("Failed to pull: {}", e);
54                log::info!("Continuing with local sync repository state...");
55            }
56        }
57    }
58
59    // Discover local sessions
60    println!("  {} local sessions...", "Discovering".cyan());
61    let local_sessions = discover_sessions(&claude_dir, &filter)?;
62    println!(
63        "  {} {} local sessions",
64        "Found".green(),
65        local_sessions.len()
66    );
67
68    // Discover remote sessions
69    let remote_projects_dir = state.sync_repo_path.join("projects");
70    println!("  {} remote sessions...", "Discovering".cyan());
71    let remote_sessions = discover_sessions(&remote_projects_dir, &filter)?;
72    println!(
73        "  {} {} remote sessions",
74        "Found".green(),
75        remote_sessions.len()
76    );
77
78    // ============================================================================
79    // SNAPSHOT CREATION: Create a snapshot before merging any changes
80    // ============================================================================
81    println!("  {} snapshot before merge...", "Creating".cyan());
82
83    // Collect all local file paths that might be affected
84    let local_file_paths: Vec<PathBuf> = local_sessions
85        .iter()
86        .map(|s| claude_dir.join(&s.file_path))
87        .collect();
88
89    // Check for large conversation files and warn users
90    warn_large_files(&local_file_paths);
91
92    // Create differential snapshot of current state (before any changes)
93    // Note: Snapshot creation failure is fatal because we need to ensure users can
94    // safely undo this pull operation if conflicts or issues occur. Without a snapshot,
95    // there would be no way to restore the previous state.
96    //
97    // Using differential snapshots saves disk space by only storing changed files.
98    let snapshot = Snapshot::create_differential(
99        OperationType::Pull,
100        local_file_paths.iter(),
101        None, // No git manager needed for pull snapshots
102    )
103    .context("Failed to create snapshot before pull")?;
104
105    // Save snapshot to disk
106    let snapshot_path = snapshot
107        .save_to_disk(None)
108        .context("Failed to save snapshot to disk")?;
109
110    if verbosity != VerbosityLevel::Quiet {
111        println!(
112            "  {} Snapshot created: {}",
113            "âœ“".green(),
114            snapshot_path
115                .file_name()
116                .map(|n| n.to_string_lossy().to_string())
117                .unwrap_or_else(|| snapshot_path.display().to_string())
118        );
119    }
120
121    // ============================================================================
122    // SHOW SUMMARY AND INTERACTIVE CONFIRMATION
123    // ============================================================================
124    if verbosity != VerbosityLevel::Quiet {
125        println!();
126        println!("{}", "Pull Summary:".bold().cyan());
127        println!("  {} Local sessions: {}", "â€¢".cyan(), local_sessions.len());
128        println!("  {} Remote sessions: {}", "â€¢".cyan(), remote_sessions.len());
129        println!();
130    }
131
132    // Show detailed file list in verbose mode
133    if verbosity == VerbosityLevel::Verbose {
134        println!("{}", "Remote sessions to be pulled:".bold());
135        for (idx, session) in remote_sessions.iter().enumerate().take(20) {
136            let relative_path = Path::new(&session.file_path)
137                .strip_prefix(&remote_projects_dir)
138                .unwrap_or(Path::new(&session.file_path));
139
140            println!("  {}. {} ({} messages)", idx + 1, relative_path.display(), session.message_count());
141        }
142        if remote_sessions.len() > 20 {
143            println!("  ... and {} more", remote_sessions.len() - 20);
144        }
145        println!();
146    }
147
148    // Interactive confirmation
149    if interactive && interactive_conflict::is_interactive() {
150        let confirm = Confirm::new("Do you want to proceed with pulling and merging these changes?")
151            .with_default(true)
152            .with_help_message("This will merge remote sessions into your local Claude Code history")
153            .prompt()
154            .context("Failed to get confirmation")?;
155
156        if !confirm {
157            println!("\n{}", "Pull cancelled.".yellow());
158            return Ok(());
159        }
160    }
161
162    // ============================================================================
163    // CONFLICT DETECTION AND RESOLUTION
164    // ============================================================================
165    if verbosity != VerbosityLevel::Quiet {
166        println!("  {} conflicts...", "Detecting".cyan());
167    }
168    let mut detector = ConflictDetector::new();
169    detector.detect(&local_sessions, &remote_sessions);
170
171    // Track affected conversations for operation record
172    let mut affected_conversations: Vec<ConversationSummary> = Vec::new();
173
174    if detector.has_conflicts() {
175        println!(
176            "  {} {} conflicts detected",
177            "!".yellow(),
178            detector.conflict_count()
179        );
180
181        // ============================================================================
182        // ATTEMPT SMART MERGE FIRST
183        // ============================================================================
184        println!("  {} smart merge...", "Attempting".cyan());
185
186        let local_map: HashMap<_, _> = local_sessions
187            .iter()
188            .map(|s| (s.session_id.clone(), s))
189            .collect();
190
191        let remote_map: HashMap<_, _> = remote_sessions
192            .iter()
193            .map(|s| (s.session_id.clone(), s))
194            .collect();
195
196        let mut smart_merge_success_count = 0;
197        let mut smart_merge_failed_conflicts = Vec::new();
198
199        for conflict in detector.conflicts_mut() {
200            // Find local and remote sessions
201            if let (Some(local_session), Some(remote_session)) = (
202                local_map.get(&conflict.session_id),
203                remote_map.get(&conflict.session_id),
204            ) {
205                // Try smart merge
206                match conflict.try_smart_merge(local_session, remote_session) {
207                    Ok(()) => {
208                        smart_merge_success_count += 1;
209                        // Write merged result to local file
210                        if let crate::conflict::ConflictResolution::SmartMerge {
211                            ref merged_entries,
212                            ref stats,
213                        } = conflict.resolution
214                        {
215                            // Create a new session with merged entries
216                            let merged_session = ConversationSession {
217                                session_id: conflict.session_id.clone(),
218                                entries: merged_entries.clone(),
219                                file_path: conflict.local_file.to_string_lossy().to_string(),
220                            };
221
222                            // Write merged session to local path
223                            if let Err(e) = merged_session.write_to_file(&conflict.local_file) {
224                                log::warn!(
225                                    "Failed to write merged session {}: {}",
226                                    conflict.session_id,
227                                    e
228                                );
229                                smart_merge_failed_conflicts.push(conflict.clone());
230                            } else {
231                                println!(
232                                    "  {} Smart merged {} ({} local + {} remote = {} total, {} branches)",
233                                    "âœ“".green(),
234                                    conflict.session_id,
235                                    stats.local_messages,
236                                    stats.remote_messages,
237                                    stats.merged_messages,
238                                    stats.branches_detected
239                                );
240                            }
241                        }
242                    }
243                    Err(e) => {
244                        log::warn!("Smart merge failed for {}: {}", conflict.session_id, e);
245                        log::info!("Falling back to manual resolution...");
246                        smart_merge_failed_conflicts.push(conflict.clone());
247                    }
248                }
249            }
250        }
251
252        println!(
253            "  {} Successfully smart merged {}/{} conflicts",
254            "âœ“".green(),
255            smart_merge_success_count,
256            detector.conflict_count()
257        );
258
259        // If some smart merges failed, handle them with interactive/keep-both resolution
260        let renames = if !smart_merge_failed_conflicts.is_empty() {
261            println!(
262                "  {} {} conflicts require manual resolution",
263                "!".yellow(),
264                smart_merge_failed_conflicts.len()
265            );
266
267            // Check if we can run interactively
268            let use_interactive = crate::interactive_conflict::is_interactive();
269
270            if use_interactive {
271                // Interactive conflict resolution for failed merges
272                println!(
273                    "\n{} Running in interactive mode for remaining conflicts",
274                    "â†’".cyan()
275                );
276
277                let resolution_result = crate::interactive_conflict::resolve_conflicts_interactive(
278                    &mut smart_merge_failed_conflicts,
279                )?;
280
281                // Apply the resolutions
282                let renames = crate::interactive_conflict::apply_resolutions(
283                    &resolution_result,
284                    &remote_sessions,
285                    &claude_dir,
286                    &remote_projects_dir,
287                )?;
288
289                // Save conflict report
290                let report = ConflictReport::from_conflicts(detector.conflicts());
291                save_conflict_report(&report)?;
292
293                renames
294            } else {
295                // Non-interactive mode: use "keep both" strategy for failed merges
296                println!(
297                    "\n{} Using automatic conflict resolution (keep both versions)",
298                    "â†’".cyan()
299                );
300
301                let mut renames = Vec::new();
302
303                println!("\n{}", "Conflict Resolution:".yellow().bold());
304                for conflict in &smart_merge_failed_conflicts {
305                    let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
306                    let conflict_suffix = format!("conflict-{timestamp}");
307
308                    if let Ok(renamed_path) = conflict.clone().resolve_keep_both(&conflict_suffix) {
309                        let relative_renamed = renamed_path
310                            .strip_prefix(&claude_dir)
311                            .unwrap_or(&renamed_path);
312                        println!(
313                            "  {} remote version saved as: {}",
314                            "â†’".yellow(),
315                            relative_renamed.display().to_string().cyan()
316                        );
317
318                        // Find and write the remote session
319                        if let Some(session) = remote_sessions
320                            .iter()
321                            .find(|s| s.session_id == conflict.session_id)
322                        {
323                            session.write_to_file(&renamed_path)?;
324                        }
325
326                        renames.push((conflict.remote_file.clone(), renamed_path));
327                    }
328                }
329
330                // Save conflict report
331                let report = ConflictReport::from_conflicts(detector.conflicts());
332                save_conflict_report(&report)?;
333
334                renames
335            }
336        } else {
337            // All conflicts resolved via smart merge
338            Vec::new()
339        };
340
341        // Track all conflicts in affected conversations
342        for (_original_path, renamed_path) in &renames {
343            let relative_path = renamed_path
344                .strip_prefix(&claude_dir)
345                .unwrap_or(renamed_path)
346                .to_string_lossy()
347                .to_string();
348
349            // Find the session ID from the renamed path
350            if let Some(session) = remote_sessions.iter().find(|s| {
351                let session_file = Path::new(&s.file_path).file_name();
352                let renamed_file = renamed_path.file_name();
353                // Try to match based on session ID in filename
354                session_file
355                    .and_then(|f| f.to_str())
356                    .and_then(|name| name.split('-').next())
357                    == renamed_file
358                        .and_then(|f| f.to_str())
359                        .and_then(|name| name.split('-').next())
360            }) {
361                match ConversationSummary::new(
362                    session.session_id.clone(),
363                    relative_path.clone(),
364                    session.latest_timestamp(),
365                    session.message_count(),
366                    SyncOperation::Conflict,
367                ) {
368                    Ok(summary) => affected_conversations.push(summary),
369                    Err(e) => log::warn!(
370                        "Failed to create summary for conflict {}: {}",
371                        relative_path,
372                        e
373                    ),
374                }
375            }
376        }
377
378        println!(
379            "\n{} View details with: claude-code-sync report",
380            "Hint:".cyan()
381        );
382    } else {
383        println!("  {} No conflicts detected", "âœ“".green());
384    }
385
386    // ============================================================================
387    // MERGE NON-CONFLICTING SESSIONS
388    // ============================================================================
389    println!("  {} non-conflicting sessions...", "Merging".cyan());
390    let local_map: HashMap<_, _> = local_sessions
391        .iter()
392        .map(|s| (s.session_id.clone(), s))
393        .collect();
394
395    let mut merged_count = 0;
396    let mut added_count = 0;
397    let mut modified_count = 0;
398    let mut unchanged_count = 0;
399
400    for remote_session in &remote_sessions {
401        // Skip if conflicts were detected
402        if detector
403            .conflicts()
404            .iter()
405            .any(|c| c.session_id == remote_session.session_id)
406        {
407            continue;
408        }
409
410        let relative_path = Path::new(&remote_session.file_path)
411            .strip_prefix(&remote_projects_dir)
412            .ok()
413            .unwrap_or_else(|| Path::new(&remote_session.file_path));
414
415        let dest_path = claude_dir.join(relative_path);
416
417        // Determine operation type based on local state
418        let operation = if let Some(local) = local_map.get(&remote_session.session_id) {
419            if local.content_hash() == remote_session.content_hash() {
420                unchanged_count += 1;
421                SyncOperation::Unchanged
422            } else {
423                modified_count += 1;
424                SyncOperation::Modified
425            }
426        } else {
427            added_count += 1;
428            SyncOperation::Added
429        };
430
431        // Copy file if it's not unchanged
432        if operation != SyncOperation::Unchanged {
433            remote_session.write_to_file(&dest_path)?;
434            merged_count += 1;
435        }
436
437        // Track all sessions (including unchanged) in affected conversations
438        let relative_path_str = relative_path.to_string_lossy().to_string();
439        match ConversationSummary::new(
440            remote_session.session_id.clone(),
441            relative_path_str.clone(),
442            remote_session.latest_timestamp(),
443            remote_session.message_count(),
444            operation,
445        ) {
446            Ok(summary) => affected_conversations.push(summary),
447            Err(e) => log::warn!("Failed to create summary for {}: {}", relative_path_str, e),
448        }
449    }
450
451    println!("  {} Merged {} sessions", "âœ“".green(), merged_count);
452
453    // ============================================================================
454    // CREATE AND SAVE OPERATION RECORD
455    // ============================================================================
456    let mut operation_record = OperationRecord::new(
457        OperationType::Pull,
458        Some(branch_name.clone()),
459        affected_conversations.clone(),
460    );
461
462    // Attach the snapshot path to the operation record
463    operation_record.snapshot_path = Some(snapshot_path);
464
465    // Load operation history and add this operation
466    let mut history = match OperationHistory::load() {
467        Ok(h) => h,
468        Err(e) => {
469            log::warn!("Failed to load operation history: {}", e);
470            log::info!("Creating new history...");
471            OperationHistory::default()
472        }
473    };
474
475    if let Err(e) = history.add_operation(operation_record) {
476        log::warn!("Failed to save operation to history: {}", e);
477        log::info!("Pull completed successfully, but history was not updated.");
478    }
479
480    // ============================================================================
481    // DISPLAY SUMMARY TO USER
482    // ============================================================================
483    println!("\n{}", "=== Pull Summary ===".bold().cyan());
484
485    // Show operation statistics
486    let conflict_count = detector.conflict_count();
487    let stats_msg = format!(
488        "  {} Added    {} Modified    {} Conflicts    {} Unchanged",
489        format!("{added_count}").green(),
490        format!("{modified_count}").cyan(),
491        format!("{conflict_count}").yellow(),
492        format!("{unchanged_count}").dimmed(),
493    );
494    println!("{stats_msg}");
495    println!();
496
497    // Group conversations by project (top-level directory)
498    let mut by_project: HashMap<String, Vec<&ConversationSummary>> = HashMap::new();
499    for conv in &affected_conversations {
500        // Skip unchanged conversations in detailed output
501        if conv.operation == SyncOperation::Unchanged {
502            continue;
503        }
504
505        let project = conv
506            .project_path
507            .split('/')
508            .next()
509            .unwrap_or("unknown")
510            .to_string();
511        by_project.entry(project).or_default().push(conv);
512    }
513
514    // Display conversations grouped by project
515    if !by_project.is_empty() {
516        println!("{}", "Affected Conversations:".bold());
517
518        let mut projects: Vec<_> = by_project.keys().collect();
519        projects.sort();
520
521        for project in projects {
522            let conversations = &by_project[project];
523            println!("\n  {} {}/", "Project:".bold(), project.cyan());
524
525            for conv in conversations.iter().take(MAX_CONVERSATIONS_TO_DISPLAY) {
526                let operation_str = match conv.operation {
527                    SyncOperation::Added => "ADD".green(),
528                    SyncOperation::Modified => "MOD".cyan(),
529                    SyncOperation::Conflict => "CONFLICT".yellow(),
530                    SyncOperation::Unchanged => "---".dimmed(),
531                };
532
533                let timestamp_str = conv
534                    .timestamp
535                    .as_ref()
536                    .and_then(|t| {
537                        // Extract just the date portion for compact display
538                        t.split('T').next()
539                    })
540                    .unwrap_or("unknown");
541
542                println!(
543                    "    {} {} ({}msg, {})",
544                    operation_str,
545                    conv.project_path,
546                    conv.message_count,
547                    timestamp_str.dimmed()
548                );
549            }
550
551            if conversations.len() > MAX_CONVERSATIONS_TO_DISPLAY {
552                println!(
553                    "    {} ... and {} more conversations",
554                    "...".dimmed(),
555                    conversations.len() - MAX_CONVERSATIONS_TO_DISPLAY
556                );
557            }
558        }
559    }
560
561    println!("\n{}", "Pull complete!".green().bold());
562
563    // Clean up old snapshots automatically
564    if let Err(e) = crate::undo::cleanup_old_snapshots(None, false) {
565        log::warn!("Failed to cleanup old snapshots: {}", e);
566    }
567
568    Ok(())
