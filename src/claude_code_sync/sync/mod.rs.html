24/// Bidirectional sync: pull remote changes, then push local changes
25pub fn sync_bidirectional(
26    commit_message: Option<&str>,
27    branch: Option<&str>,
28    exclude_attachments: bool,
29    interactive: bool,
30    verbosity: crate::VerbosityLevel,
31) -> Result<()> {
32    use crate::VerbosityLevel;
33
34    if verbosity != VerbosityLevel::Quiet {
35        println!("{}", "=== Bidirectional Sync ===".bold().cyan());
36        println!();
37        println!("{}", "Step 1: Pulling remote changes...".bold());
38    }
39
40    // First, pull remote changes
41    pull_history(true, branch, interactive, verbosity)?;
42
43    if verbosity != VerbosityLevel::Quiet {
44        println!();
45        println!("{}", "Step 2: Pushing local changes...".bold());
46    }
47
48    // Then, push local changes
49    push_history(commit_message, true, branch, exclude_attachments, interactive, verbosity)?;
50
51    if verbosity == VerbosityLevel::Quiet {
52        println!("Sync complete");
53    } else {
54        println!();
55        println!("{}", "=== Sync Complete ===".green().bold());
56        println!(
57            "  {} Your local and remote histories are now in sync",
58            "âœ“".green()
59        );
60    }
61
62    Ok(())
63}

73    #[test]
74    fn test_url_validation() {
75        let temp_dir = TempDir::new().unwrap();
76        let repo_path = temp_dir.path().join("test-repo");
77
78        // Initialize a test repo
79        GitManager::init(&repo_path).unwrap();
80
81        // Save a test state
82        let state = SyncState {
83            sync_repo_path: repo_path.clone(),
84            has_remote: false,
85            is_cloned_repo: false,
86        };
87
88        // Create state directory using ConfigManager
89        let _state_path = crate::config::ConfigManager::ensure_config_dir().unwrap();
90        let state_file = crate::config::ConfigManager::state_file_path().unwrap();
91        std::fs::write(&state_file, serde_json::to_string(&state).unwrap()).unwrap();
92
93        // Valid HTTPS URL
94        let result = set_remote("origin", "https://github.com/user/repo.git");
95        assert!(result.is_ok());
96
97        // Valid HTTP URL
98        let result = set_remote("origin", "http://gitlab.com/user/repo.git");
99        assert!(result.is_ok());
100
101        // Valid SSH URL
102        let result = set_remote("origin", "git@github.com:user/repo.git");
103        assert!(result.is_ok());
104
105        // Invalid URL (missing protocol)
106        let result = set_remote("origin", "github.com/user/repo.git");
107        assert!(result.is_err());
108        if let Err(e) = result {
109            let error_msg = e.to_string();
110            assert!(error_msg.contains("Invalid URL format"));
111        }
112
113        // Cleanup
114        std::fs::remove_file(&state_file).ok();
115    }

117    #[test]
118    fn test_filter_with_attachments() {
119        let filter = FilterConfig {
120            exclude_attachments: true,
121            ..Default::default()
122        };
123
124        // JSONL files should be included
125        assert!(filter.should_include(Path::new("session.jsonl")));
126        assert!(filter.should_include(Path::new("/path/to/session.jsonl")));
127
128        // Non-JSONL files should be excluded
129        assert!(!filter.should_include(Path::new("image.png")));
130        assert!(!filter.should_include(Path::new("document.pdf")));
131        assert!(!filter.should_include(Path::new("archive.zip")));
132        assert!(!filter.should_include(Path::new("/path/to/file.jpg")));
133    }
