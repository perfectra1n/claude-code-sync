20/// Push local Claude Code history to sync repository
21pub fn push_history(
22    commit_message: Option<&str>,
23    push_remote: bool,
24    branch: Option<&str>,
25    exclude_attachments: bool,
26    interactive: bool,
27    verbosity: crate::VerbosityLevel,
28) -> Result<()> {
29    use crate::VerbosityLevel;
30
31    if verbosity != VerbosityLevel::Quiet {
32        println!("{}", "Pushing Claude Code history...".cyan().bold());
33    }
34
35    let state = SyncState::load()?;
36    let git_manager = GitManager::open(&state.sync_repo_path)?;
37    let mut filter = FilterConfig::load()?;
38
39    // Override exclude_attachments if specified in command
40    if exclude_attachments {
41        filter.exclude_attachments = true;
42    }
43    let claude_dir = claude_projects_dir()?;
44
45    // Get the current branch name for operation record
46    let branch_name = branch
47        .map(|s| s.to_string())
48        .or_else(|| git_manager.current_branch().ok())
49        .unwrap_or_else(|| "main".to_string());
50
51    // Discover all sessions
52    println!("  {} conversation sessions...", "Discovering".cyan());
53    let sessions = discover_sessions(&claude_dir, &filter)?;
54    println!("  {} {} sessions", "Found".green(), sessions.len());
55
56    // ============================================================================
57    // COPY SESSIONS AND TRACK CHANGES
58    // ============================================================================
59    let projects_dir = state.sync_repo_path.join("projects");
60    fs::create_dir_all(&projects_dir)?;
61
62    // Discover existing sessions in sync repo to determine operation type
63    println!("  {} sessions to sync repository...", "Copying".cyan());
64    let existing_sessions = discover_sessions(&projects_dir, &filter)?;
65    let existing_map: HashMap<_, _> = existing_sessions
66        .iter()
67        .map(|s| (s.session_id.clone(), s))
68        .collect();
69
70    // Track pushed conversations for operation record
71    let mut pushed_conversations: Vec<ConversationSummary> = Vec::new();
72    let mut added_count = 0;
73    let mut modified_count = 0;
74    let mut unchanged_count = 0;
75
76    for session in &sessions {
77        let relative_path = Path::new(&session.file_path)
78            .strip_prefix(&claude_dir)
79            .unwrap_or(Path::new(&session.file_path));
80
81        let dest_path = projects_dir.join(relative_path);
82
83        // Determine operation type based on existing state
84        let operation = if let Some(existing) = existing_map.get(&session.session_id) {
85            if existing.content_hash() == session.content_hash() {
86                unchanged_count += 1;
87                SyncOperation::Unchanged
88            } else {
89                modified_count += 1;
90                SyncOperation::Modified
91            }
92        } else {
93            added_count += 1;
94            SyncOperation::Added
95        };
96
97        // Write the session file
98        session.write_to_file(&dest_path)?;
99
100        // Track this session in pushed conversations
101        let relative_path_str = relative_path.to_string_lossy().to_string();
102        match ConversationSummary::new(
103            session.session_id.clone(),
104            relative_path_str.clone(),
105            session.latest_timestamp(),
106            session.message_count(),
107            operation,
108        ) {
109            Ok(summary) => pushed_conversations.push(summary),
110            Err(e) => log::warn!(
111                "Failed to create summary for {}: {}",
112                relative_path_str,
113                e
114            ),
115        }
116    }
117
118    // ============================================================================
119    // SHOW SUMMARY AND INTERACTIVE CONFIRMATION
120    // ============================================================================
121    if verbosity != VerbosityLevel::Quiet {
122        println!();
123        println!("{}", "Push Summary:".bold().cyan());
124        println!("  {} Added: {}", "•".green(), added_count);
125        println!("  {} Modified: {}", "•".yellow(), modified_count);
126        println!("  {} Unchanged: {}", "•".dimmed(), unchanged_count);
127        println!("  {} Total: {}", "•".cyan(), sessions.len());
128        println!();
129    }
130
131    // Show detailed file list in verbose mode
132    if verbosity == VerbosityLevel::Verbose {
133        println!("{}", "Files to be pushed:".bold());
134        for (idx, session) in sessions.iter().enumerate().take(20) {
135            let relative_path = Path::new(&session.file_path)
136                .strip_prefix(&claude_dir)
137                .unwrap_or(Path::new(&session.file_path));
138
139            let status = if let Some(existing) = existing_map.get(&session.session_id) {
140                if existing.content_hash() == session.content_hash() {
141                    "unchanged".dimmed()
142                } else {
143                    "modified".yellow()
144                }
145            } else {
146                "new".green()
147            };
148
149            println!("  {}. {} [{}]", idx + 1, relative_path.display(), status);
150        }
151        if sessions.len() > 20 {
152            println!("  ... and {} more", sessions.len() - 20);
153        }
154        println!();
155    }
156
157    // Interactive confirmation
158    if interactive && interactive_conflict::is_interactive() {
159        let confirm = Confirm::new("Do you want to proceed with pushing these changes?")
160            .with_default(true)
161            .with_help_message("This will commit and push to the sync repository")
162            .prompt()
163            .context("Failed to get confirmation")?;
164
165        if !confirm {
166            println!("\n{}", "Push cancelled.".yellow());
167            return Ok(());
168        }
169    }
170
171    // ============================================================================
172    // COMMIT AND PUSH CHANGES
173    // ============================================================================
174    git_manager.stage_all()?;
175
176    let has_changes = git_manager.has_changes()?;
177    if has_changes {
178        // ============================================================================
179        // SNAPSHOT CREATION: Create a snapshot before committing changes
180        // ============================================================================
181        if verbosity != VerbosityLevel::Quiet {
182            println!("  {} snapshot before push...", "Creating".cyan());
183        }
184
185        // Get the current commit hash before making any changes
186        // This allows us to undo the push later by resetting to this commit
187        let commit_before_push = git_manager
188            .current_commit_hash()
189            .context("Failed to get current commit hash")?;
190
191        // Collect all file paths in the sync repository that will be affected
192        // For push operations, we snapshot the sync repository state, not local files
193        let sync_repo_files: Vec<PathBuf> = sessions
194            .iter()
195            .map(|s| {
196                let relative_path = Path::new(&s.file_path)
197                    .strip_prefix(&claude_dir)
198                    .unwrap_or(Path::new(&s.file_path));
199                state.sync_repo_path.join("projects").join(relative_path)
200            })
201            .collect();
202
203        // Check for large conversation files and warn users
204        warn_large_files(&sync_repo_files);
205
206        // Create differential snapshot of sync repository state before push
207        // Note: Snapshot creation failure is fatal because we need to ensure users can
208        // safely undo this push operation if issues occur. Without a snapshot,
209        // there would be no way to restore the previous repository state.
210        //
211        // Using differential snapshots saves disk space by only storing changed files.
212        let snapshot = Snapshot::create_differential(
213            OperationType::Push,
214            sync_repo_files.iter(),
215            Some(&git_manager), // Pass git manager to capture commit hash
216        )
217        .context("Failed to create snapshot before push")?;
218
219        // Save snapshot to disk
220        let snapshot_path = snapshot
221            .save_to_disk(None)
222            .context("Failed to save snapshot to disk")?;
223
224        println!(
225            "  {} Snapshot created: {} (commit: {})",
226            "✓".green(),
227            snapshot_path
228                .file_name()
229                .map(|n| n.to_string_lossy().to_string())
230                .unwrap_or_else(|| snapshot_path.display().to_string()),
231            &commit_before_push[..8]
232        );
233
234        let default_message = format!(
235            "Sync {} sessions at {}",
236            sessions.len(),
237            chrono::Utc::now().format("%Y-%m-%d %H:%M:%S UTC")
238        );
239        let message = commit_message.unwrap_or(&default_message);
240
241        println!("  {} changes...", "Committing".cyan());
242        git_manager.commit(message)?;
243        println!("  {} Committed: {}", "✓".green(), message);
244
245        // Push to remote if configured
246        if push_remote && state.has_remote {
247            println!("  {} to remote...", "Pushing".cyan());
248
249            match git_manager.push("origin", &branch_name) {
250                Ok(_) => println!("  {} Pushed to origin/{}", "✓".green(), branch_name),
251                Err(e) => log::warn!("Failed to push: {}", e),
252            }
253        }
254
255        // ============================================================================
256        // CREATE AND SAVE OPERATION RECORD
257        // ============================================================================
258        let mut operation_record = OperationRecord::new(
259            OperationType::Push,
260            Some(branch_name.clone()),
261            pushed_conversations.clone(),
262        );
263
264        // Attach the snapshot path to the operation record
265        operation_record.snapshot_path = Some(snapshot_path);
266
267        // Load operation history and add this operation
268        let mut history = match OperationHistory::load() {
269            Ok(h) => h,
270            Err(e) => {
271                log::warn!("Failed to load operation history: {}", e);
272                log::info!("Creating new history...");
273                OperationHistory::default()
274            }
275        };
276
277        if let Err(e) = history.add_operation(operation_record) {
278            log::warn!("Failed to save operation to history: {}", e);
279            log::info!("Push completed successfully, but history was not updated.");
280        }
281    } else {
282        println!("  {} No changes to commit", "Note:".yellow());
283    }
284
285    // ============================================================================
286    // DISPLAY SUMMARY TO USER
287    // ============================================================================
288    println!("\n{}", "=== Push Summary ===".bold().cyan());
289
290    // Show operation statistics
291    let stats_msg = format!(
292        "  {} Added    {} Modified    {} Unchanged",
293        format!("{added_count}").green(),
294        format!("{modified_count}").cyan(),
295        format!("{unchanged_count}").dimmed(),
296    );
297    println!("{stats_msg}");
298    println!();
299
300    // Group conversations by project (top-level directory)
301    let mut by_project: HashMap<String, Vec<&ConversationSummary>> = HashMap::new();
302    for conv in &pushed_conversations {
303        // Skip unchanged conversations in detailed output
304        if conv.operation == SyncOperation::Unchanged {
305            continue;
306        }
307
308        let project = conv
309            .project_path
310            .split('/')
311            .next()
312            .unwrap_or("unknown")
313            .to_string();
314        by_project.entry(project).or_default().push(conv);
315    }
316
317    // Display conversations grouped by project
318    if !by_project.is_empty() {
319        println!("{}", "Pushed Conversations:".bold());
320
321        let mut projects: Vec<_> = by_project.keys().collect();
322        projects.sort();
323
324        for project in projects {
325            let conversations = &by_project[project];
326            println!("\n  {} {}/", "Project:".bold(), project.cyan());
327
328            for conv in conversations.iter().take(MAX_CONVERSATIONS_TO_DISPLAY) {
329                let operation_str = match conv.operation {
330                    SyncOperation::Added => "ADD".green(),
331                    SyncOperation::Modified => "MOD".cyan(),
332                    SyncOperation::Conflict => "CONFLICT".yellow(),
333                    SyncOperation::Unchanged => "---".dimmed(),
334                };
335
336                let timestamp_str = conv
337                    .timestamp
338                    .as_ref()
339                    .and_then(|t| {
340                        // Extract just the date portion for compact display
341                        t.split('T').next()
342                    })
343                    .unwrap_or("unknown");
344
345                println!(
346                    "    {} {} ({}msg, {})",
347                    operation_str,
348                    conv.project_path,
349                    conv.message_count,
350                    timestamp_str.dimmed()
351                );
352            }
353
354            if conversations.len() > MAX_CONVERSATIONS_TO_DISPLAY {
355                println!(
356                    "    {} ... and {} more conversations",
357                    "...".dimmed(),
358                    conversations.len() - MAX_CONVERSATIONS_TO_DISPLAY
359                );
360            }
361        }
362    }
363
364    if verbosity == VerbosityLevel::Quiet {
365        println!("Push complete");
366    } else {
367        println!("\n{}", "Push complete!".green().bold());
368    }
369
370    // Clean up old snapshots automatically
371    if let Err(e) = crate::undo::cleanup_old_snapshots(None, false) {
372        log::warn!("Failed to cleanup old snapshots: {}", e);
373    }
374
375    Ok(())
376}
