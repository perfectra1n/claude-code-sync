8    /// Restore files from this snapshot
9    ///
10    /// This writes all files from the snapshot back to their original locations,
11    /// overwriting any current content.
12    ///
13    /// For differential snapshots, this recursively loads the base snapshot chain
14    /// and applies all changes in order to reconstruct the full state.
15    ///
16    /// # Security
17    /// This method validates all paths to prevent path traversal attacks.
18    /// By default, only paths within the home directory are allowed.
19    /// For testing, you can pass a custom allowed_base_dir.
20    ///
21    /// # Arguments
22    /// * `allowed_base_dir` - Optional base directory for path validation.
23    ///   If None, defaults to home directory for security.
24    /// * `snapshots_dir` - Optional snapshots directory (for testing with differential snapshots)
25    pub fn restore_with_base_and_snapshots(
26        &self,
27        allowed_base_dir: Option<&Path>,
28        snapshots_dir: Option<&Path>,
29    ) -> Result<()> {
30        // Determine the allowed base directory
31        let allowed_base = if let Some(base) = allowed_base_dir {
32            // For testing: use the provided base
33            base.canonicalize().with_context(|| {
34                format!("Failed to canonicalize base directory: {}", base.display())
35            })?
36        } else {
37            // For production: use home directory
38            let home_dir = dirs::home_dir().context("Failed to get home directory")?;
39            home_dir
40                .canonicalize()
41                .context("Failed to canonicalize home directory")?
42        };
43
44        // Build the complete file state by walking the snapshot chain
45        let all_files = self.reconstruct_full_state_with_dir(snapshots_dir)?;
46
47        // First, handle file deletions from the snapshot
48        for deleted_path in &self.deleted_files {
49            let path = PathBuf::from(deleted_path);
50
51            // Validate the path is within allowed directory
52            if let Ok(canonical) = path.canonicalize() {
53                if canonical.starts_with(&allowed_base) && path.exists() {
54                    fs::remove_file(&path).with_context(|| {
55                        format!("Failed to delete file: {}", path.display())
56                    })?;
57                }
58            }
59        }
60
61        // Then restore all files from the reconstructed state
62        for (path_str, content) in &all_files {
63            let path = PathBuf::from(path_str);
64
65            // Canonicalize the path to resolve any symlinks or .. components
66            // First ensure parent directory exists for canonicalization to work
67            if let Some(parent) = path.parent() {
68                fs::create_dir_all(parent)
69                    .with_context(|| format!("Failed to create directory: {}", parent.display()))?;
70            }
71
72            // Create the file if it doesn't exist for canonicalization
73            if !path.exists() {
74                fs::write(&path, b"").with_context(|| {
75                    format!("Failed to create temporary file: {}", path.display())
76                })?;
77            }
78
79            let canonical_path = path
80                .canonicalize()
81                .with_context(|| format!("Failed to canonicalize path: {}", path.display()))?;
82
83            // Validate the canonical path is within the allowed base directory
84            if !canonical_path.starts_with(&allowed_base) {
85                return Err(anyhow!(
86                    "Security: Path traversal detected. Path {} is outside allowed directory {}",
87                    path.display(),
88                    allowed_base.display()
89                ));
90            }
91
92            // Now write the actual content
93            fs::write(&canonical_path, content)
94                .with_context(|| format!("Failed to restore file: {}", canonical_path.display()))?;
95        }
96
97        Ok(())
98    }
99
100    /// Restore files from this snapshot using default snapshots directory
101    ///
102    /// This is a wrapper around `restore_with_base_and_snapshots` for backwards compatibility.
103    pub fn restore_with_base(&self, allowed_base_dir: Option<&Path>) -> Result<()> {
104        self.restore_with_base_and_snapshots(allowed_base_dir, None)
105    }
106
107    /// Restore files from this snapshot
108    ///
109    /// This is a convenience wrapper that uses the home directory as the allowed base.
110    #[allow(dead_code)]
111    pub fn restore(&self) -> Result<()> {
112        self.restore_with_base(None)
113    }
