11/// Represents a snapshot of conversation files at a point in time
12///
13/// Snapshots are created before each sync operation to enable undo functionality.
14/// They capture the complete state of all conversation files that might be affected.
15///
16/// ## Differential Snapshots
17///
18/// To save disk space, snapshots can be differential - only storing files that changed
19/// since the previous snapshot. This is controlled by the `base_snapshot_id` field:
20/// - `None`: Full snapshot containing all files
21/// - `Some(id)`: Differential snapshot containing only changes since base snapshot
22///
23/// When restoring a differential snapshot, we recursively load the chain of base
24/// snapshots to reconstruct the full state.
25#[derive(Debug, Clone, Serialize, Deserialize)]
26pub struct Snapshot {
27    /// Unique identifier for this snapshot
28    pub snapshot_id: String,
29
30    /// When this snapshot was created
31    pub timestamp: chrono::DateTime<chrono::Utc>,
32
33    /// Type of operation this snapshot was created for
34    pub operation_type: OperationType,
35
36    /// Git commit hash before the operation (for push operations)
37    #[serde(skip_serializing_if = "Option::is_none")]
38    pub git_commit_hash: Option<String>,
39
40    /// Mapping of file paths (relative to Claude projects dir) to their content
41    ///
42    /// We store the raw bytes to preserve exact file state including encoding.
43    /// The HashMap key is a string path for JSON serialization compatibility.
44    ///
45    /// For differential snapshots, only contains files that changed/were added.
46    #[serde(with = "base64_map")]
47    pub files: HashMap<String, Vec<u8>>,
48
49    /// Git branch name at the time of snapshot
50    #[serde(skip_serializing_if = "Option::is_none")]
51    pub branch: Option<String>,
52
53    /// Base snapshot ID for differential snapshots
54    ///
55    /// If present, this snapshot only contains changes relative to the base.
56    /// The full state can be reconstructed by loading the chain of snapshots.
57    #[serde(skip_serializing_if = "Option::is_none")]
58    pub base_snapshot_id: Option<String>,
59
60    /// Files that were deleted since the base snapshot
61    ///
62    /// Only populated for differential snapshots. Lists file paths that existed
63    /// in the base snapshot but should be removed when restoring this snapshot.
64    #[serde(default, skip_serializing_if = "Vec::is_empty")]
65    pub deleted_files: Vec<String>,
66}

68/// Custom serialization for `HashMap<String, Vec<u8>>` using base64 encoding
69///
70/// This is necessary because JSON doesn't natively support binary data,
71/// so we encode file contents as base64 strings for storage.
72mod base64_map {
73    use base64::engine::general_purpose::STANDARD;
74    use base64::Engine;
75    use serde::{Deserialize, Deserializer, Serialize, Serializer};
76    use std::collections::HashMap;
77
78    pub fn serialize<S>(map: &HashMap<String, Vec<u8>>, serializer: S) -> Result<S::Ok, S::Error>
79    where
80        S: Serializer,
81    {
82        let base64_map: HashMap<String, String> = map
83            .iter()
84            .map(|(k, v)| (k.clone(), STANDARD.encode(v)))
85            .collect();
86        base64_map.serialize(serializer)
87    }
88
89    pub fn deserialize<'de, D>(deserializer: D) -> Result<HashMap<String, Vec<u8>>, D::Error>
90    where
91        D: Deserializer<'de>,
92    {
93        let base64_map: HashMap<String, String> = HashMap::deserialize(deserializer)?;
94        base64_map
95            .into_iter()
96            .map(|(k, v)| {
97                STANDARD
98                    .decode(&v)
99                    .map(|bytes| (k, bytes))
100                    .map_err(serde::de::Error::custom)
101            })
102            .collect()
103    }
104}

106impl Snapshot {
107    /// Create a new snapshot from a set of file paths
108    ///
109    /// # Arguments
110    /// * `operation_type` - Type of operation this snapshot is for
111    /// * `file_paths` - Iterator of file paths to include in snapshot
112    /// * `commit_hash` - Optional git commit hash to store in the snapshot
113    ///
114    /// # Returns
115    /// A new Snapshot instance with all file contents captured
116    pub fn create<P, I>(
117        operation_type: OperationType,
118        file_paths: I,
119        commit_hash: Option<&str>,
120    ) -> Result<Self>
121    where
122        P: AsRef<Path>,
123        I: IntoIterator<Item = P>,
124    {
125        let snapshot_id = Uuid::new_v4().to_string();
126        let timestamp = chrono::Utc::now();
127        let mut files = HashMap::new();
128
129        // Capture current state of all specified files
130        for path in file_paths {
131            let path = path.as_ref();
132
133            // Use direct read instead of checking existence first to avoid TOCTOU
134            match fs::read(path) {
135                Ok(content) => {
136                    // Store with path as string for JSON compatibility
137                    files.insert(path.to_string_lossy().to_string(), content);
138                }
139                Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
140                    // File doesn't exist - this is expected in some cases (e.g., deleted files)
141                    // Just skip it without error
142                    continue;
143                }
144                Err(e) => {
145                    // Other errors should be reported
146                    return Err(e).with_context(|| {
147                        format!("Failed to read file for snapshot: {}", path.display())
148                    });
149                }
150            }
151        }
152
153        Ok(Snapshot {
154            snapshot_id,
155            timestamp,
156            operation_type,
157            git_commit_hash: commit_hash.map(|s| s.to_string()),
158            files,
159            branch: None,
160            base_snapshot_id: None,
161            deleted_files: Vec::new(),
162        })
163    }
164
165    /// Save this snapshot to disk
166    ///
167    /// Snapshots are saved to `~/.claude-code-sync/snapshots/{snapshot_id}.json`
168    ///
169    /// # Arguments
170    /// * `custom_path` - Optional custom directory to save snapshot (for testing)
171    pub fn save_to_disk(&self, custom_path: Option<&Path>) -> Result<PathBuf> {
172        let snapshot_dir = if let Some(path) = custom_path {
173            path.to_path_buf()
174        } else {
175            Self::snapshots_dir()?
176        };
177
178        // Ensure snapshots directory exists
179        fs::create_dir_all(&snapshot_dir).with_context(|| {
180            format!(
181                "Failed to create snapshots directory: {}",
182                snapshot_dir.display()
183            )
184        })?;
185
186        let snapshot_path = snapshot_dir.join(format!("{}.json", self.snapshot_id));
187
188        let json =
189            serde_json::to_string_pretty(self).context("Failed to serialize snapshot to JSON")?;
190
191        fs::write(&snapshot_path, &json).with_context(|| {
192            format!(
193                "Failed to write snapshot to disk: {}",
194                snapshot_path.display()
195            )
196        })?;
197
198        // Log snapshot size information (to file only, UI output is handled by caller)
199        let size_mb = json.len() as f64 / (1024.0 * 1024.0);
200        let snapshot_type = if self.base_snapshot_id.is_some() {
201            "differential"
202        } else {
203            "full"
204        };
205
206        log::debug!(
207            "Created {} snapshot: {} ({:.1} MB, {} files)",
208            snapshot_type,
209            self.snapshot_id,
210            size_mb,
211            self.files.len()
212        );
213
214        if size_mb > 100.0 {
215            log::debug!("Large snapshot size - consider cleaning up old conversation files");
216        }
217
218        Ok(snapshot_path)
219    }
220
221    /// Load a snapshot from disk
222    ///
223    /// # Arguments
224    /// * `snapshot_path` - Path to the snapshot JSON file
225    ///
226    /// # Behavior
227    /// - Logs snapshot size information
228    /// - Warns if snapshot is unusually large (>100MB) but still loads it
229    /// - Snapshots are critical for undo functionality and should never be skipped
230    ///
231    /// # Errors
232    /// Returns error only if file cannot be read or parsed
233    pub fn load_from_disk<P: AsRef<Path>>(snapshot_path: P) -> Result<Self> {
234        let snapshot_path = snapshot_path.as_ref();
235
236        // Get file size for logging
237        let metadata = fs::metadata(snapshot_path).with_context(|| {
238            format!(
239                "Failed to read snapshot file metadata: {}",
240                snapshot_path.display()
241            )
242        })?;
243
244        let size_mb = metadata.len() as f64 / (1024.0 * 1024.0);
245
246        // Log size information - always show this for visibility
247        if size_mb > 100.0 {
248            println!(
249                "    {} Loading large snapshot: {} ({:.1} MB) - This may take a moment...",
250                "⚠".yellow(),
251                snapshot_path.file_name().unwrap().to_string_lossy().cyan(),
252                size_mb
253            );
254        } else {
255            println!(
256                "    {} snapshot: {} ({:.1} MB)",
257                "Loading".dimmed(),
258                snapshot_path.file_name().unwrap().to_string_lossy().cyan(),
259                size_mb
260            );
261        }
262
263        let content = fs::read_to_string(snapshot_path).with_context(|| {
264            format!("Failed to read snapshot file: {}", snapshot_path.display())
265        })?;
266
267        let snapshot: Snapshot = serde_json::from_str(&content).with_context(|| {
268            format!("Failed to parse snapshot JSON: {}", snapshot_path.display())
269        })?;
270
271        // Log file count information
272        println!("    {} {} files", "Contains".dimmed(), snapshot.files.len());
273
274        // Warn if unusually large number of files
275        if snapshot.files.len() > 1000 {
276            println!(
277                "    {} Large number of files - this is a full (non-differential) snapshot",
278                "Note:".yellow()
279            );
280        }
281
282        Ok(snapshot)
283    }
284
285    /// Get the default snapshots directory
286    pub(crate) fn snapshots_dir() -> Result<PathBuf> {
287        crate::config::ConfigManager::snapshots_dir()
288    }
