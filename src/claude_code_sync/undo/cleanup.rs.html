9/// Configuration for snapshot cleanup
10pub struct SnapshotCleanupConfig {
11    /// Keep at most this many snapshots per operation type (pull/push)
12    pub max_count_per_type: usize,
13    /// Keep snapshots newer than this many days
14    pub max_age_days: i64,
15}
16
17impl Default for SnapshotCleanupConfig {
18    fn default() -> Self {
19        Self {
20            max_count_per_type: 5,
21            max_age_days: 7,
22        }
23    }
24}

26/// Clean up old snapshots based on age and count limits
27///
28/// This removes snapshots that don't meet EITHER of the following criteria:
29/// - Within the last N snapshots of each operation type
30/// - Created within the last X days
31///
32/// # Arguments
33/// * `config` - Cleanup configuration (defaults: keep last 5 per type, last 7 days)
34/// * `dry_run` - If true, show what would be deleted without actually deleting
35/// * `snapshots_dir` - Optional custom snapshots directory (for testing)
36///
37/// # Returns
38/// Number of snapshots deleted
39pub fn cleanup_old_snapshots_with_dir(
40    config: Option<SnapshotCleanupConfig>,
41    dry_run: bool,
42    snapshots_dir: Option<&Path>,
43) -> Result<usize> {
44    let config = config.unwrap_or_default();
45    let snapshots_dir = if let Some(dir) = snapshots_dir {
46        dir.to_path_buf()
47    } else {
48        Snapshot::snapshots_dir()?
49    };
50
51    if !snapshots_dir.exists() {
52        return Ok(0);
53    }
54
55    // Collect all snapshots with metadata
56    let mut pull_snapshots: Vec<(std::path::PathBuf, chrono::DateTime<chrono::Utc>)> = Vec::new();
57    let mut push_snapshots: Vec<(std::path::PathBuf, chrono::DateTime<chrono::Utc>)> = Vec::new();
58
59    for entry in fs::read_dir(&snapshots_dir)? {
60        let entry = entry?;
61        let path = entry.path();
62
63        if !path.extension().map_or(false, |ext| ext == "json") {
64            continue;
65        }
66
67        // Load snapshot metadata
68        if let Ok(content) = fs::read_to_string(&path) {
69            if let Ok(snapshot) = serde_json::from_str::<Snapshot>(&content) {
70                match snapshot.operation_type {
71                    OperationType::Pull => pull_snapshots.push((path, snapshot.timestamp)),
72                    OperationType::Push => push_snapshots.push((path, snapshot.timestamp)),
73                }
74            }
75        }
76    }
77
78    // Sort by timestamp descending (newest first)
79    pull_snapshots.sort_by(|a, b| b.1.cmp(&a.1));
80    push_snapshots.sort_by(|a, b| b.1.cmp(&a.1));
81
82    // Determine which snapshots to keep
83    let now = chrono::Utc::now();
84    let age_threshold = now - chrono::Duration::days(config.max_age_days);
85
86    let mut to_delete = Vec::new();
87
88    // Process pull snapshots
89    for (idx, (path, timestamp)) in pull_snapshots.iter().enumerate() {
90        let within_count_limit = idx < config.max_count_per_type;
91        let within_age_limit = *timestamp >= age_threshold;
92
93        // Delete if it doesn't meet EITHER criterion
94        if !within_count_limit && !within_age_limit {
95            to_delete.push(path.clone());
96        }
97    }
98
99    // Process push snapshots
100    for (idx, (path, timestamp)) in push_snapshots.iter().enumerate() {
101        let within_count_limit = idx < config.max_count_per_type;
102        let within_age_limit = *timestamp >= age_threshold;
103
104        if !within_count_limit && !within_age_limit {
105            to_delete.push(path.clone());
106        }
107    }
108
109    // Delete the snapshots (or just report in dry run mode)
110    let deleted_count = to_delete.len();
111
112    if dry_run {
113        println!("Would delete {} snapshots:", deleted_count);
114        for path in &to_delete {
115            println!("  - {}", path.display());
116        }
117    } else {
118        for path in &to_delete {
119            if let Err(e) = fs::remove_file(path) {
120                warn!("Failed to delete snapshot {}: {}", path.display(), e);
121            }
122        }
123    }
124
125    Ok(deleted_count)
126}

128/// Clean up old snapshots using the default snapshots directory
129///
130/// This is a convenience wrapper around `cleanup_old_snapshots_with_dir`.
131///
132/// # Arguments
133/// * `config` - Cleanup configuration (defaults: keep last 5 per type, last 7 days)
134/// * `dry_run` - If true, show what would be deleted without actually deleting
135///
136/// # Returns
137/// Number of snapshots deleted
138pub fn cleanup_old_snapshots(
139    config: Option<SnapshotCleanupConfig>,
140    dry_run: bool,
141) -> Result<usize> {
142    cleanup_old_snapshots_with_dir(config, dry_run, None)
