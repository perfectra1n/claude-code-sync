8/// Initialize the logging system
9///
10/// Sets up logging to both console and a log file in the config directory.
11///
12/// **Console logging** can be controlled via the `RUST_LOG` environment variable:
13/// - `RUST_LOG=error` - Only errors
14/// - `RUST_LOG=warn` - Warnings and errors
15/// - `RUST_LOG=info` - Info, warnings, and errors (default)
16/// - `RUST_LOG=debug` - Debug and above
17/// - `RUST_LOG=trace` - Everything
18///
19/// **File logging** always captures all levels and is stored at:
20/// - Linux: ~/.config/claude-code-sync/claude-code-sync.log or $XDG_CONFIG_HOME/claude-code-sync/claude-code-sync.log
21/// - macOS: ~/Library/Application Support/claude-code-sync/claude-code-sync.log
22/// - Windows: %APPDATA%\claude-code-sync\claude-code-sync.log
23///
24/// ## Examples
25///
26/// ```bash
27/// # Show all debug messages on console
28/// RUST_LOG=debug claude-code-sync sync
29///
30/// # Only show errors on console
31/// RUST_LOG=error claude-code-sync push
32///
33/// # No console output (file logging continues)
34/// RUST_LOG=off claude-code-sync pull
35/// ```
36pub fn init_logger() -> Result<()> {
37    // Ensure config directory exists
38    ConfigManager::ensure_config_dir()?;
39
40    // Determine if console logging should be enabled
41    // By default, use Info level unless RUST_LOG is set
42    let default_level = std::env::var("RUST_LOG")
43        .ok()
44        .and_then(|s| s.parse::<LevelFilter>().ok())
45        .unwrap_or(LevelFilter::Info);
46
47    // Initialize env_logger with custom format
48    env_logger::Builder::from_default_env()
49        .format(|buf, record| {
50            writeln!(
51                buf,
52                "{} [{:5}] {}",
53                chrono::Local::now().format("%H:%M:%S"),
54                record.level(),
55                record.args()
56            )
57        })
58        .filter_level(default_level)
59        .target(env_logger::Target::Stdout)
60        .try_init()
61        .ok(); // Ignore error if logger is already initialized
62
63    // Also log initialization to file
64    log_to_file(&format!(
65        "Logger initialized with level: {default_level:?}"
66    ))?;
67
68    Ok(())
69}

71/// Log to file only (useful for background operations or detailed logging)
72pub fn log_to_file(message: &str) -> Result<()> {
73    let log_path = ConfigManager::log_file_path()?;
74
75    let mut file = OpenOptions::new()
76        .create(true)
77        .append(true)
78        .open(&log_path)
79        .with_context(|| format!("Failed to open log file: {}", log_path.display()))?;
80
81    writeln!(
82        file,
83        "[{}] {}",
84        chrono::Local::now().format("%Y-%m-%d %H:%M:%S"),
85        message
86    )?;
87
88    Ok(())
89}

91/// Rotate log file if it exceeds the size limit (default: 10MB)
92pub fn rotate_log_if_needed() -> Result<()> {
93    const MAX_LOG_SIZE: u64 = 10 * 1024 * 1024; // 10MB
94
95    let log_path = ConfigManager::log_file_path()?;
96
97    // Check if log file exists and its size
98    if log_path.exists() {
99        let metadata = std::fs::metadata(&log_path)?;
100
101        if metadata.len() > MAX_LOG_SIZE {
102            // Rotate: rename current log to .old and start fresh
103            let old_log_path = log_path.with_extension("log.old");
104
105            // Remove old backup if it exists
106            if old_log_path.exists() {
107                std::fs::remove_file(&old_log_path)?;
108            }
109
110            // Rename current log to .old
111            std::fs::rename(&log_path, &old_log_path)?;
112
113            log::info!("Log file rotated to {}", old_log_path.display());
114        }
115    }
116
117    Ok(())
118}
