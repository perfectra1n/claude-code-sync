8/// Represents a conflict between local and remote versions of the same conversation session.
9///
10/// A `Conflict` is detected when both local and remote filesystems contain a conversation
11/// with the same session ID but different content (as determined by content hashes). This
12/// typically occurs when the same conversation has been modified on different machines or
13/// when changes haven't been synchronized properly.
14///
15/// The conflict contains metadata about both versions to help users make informed decisions
16/// about how to resolve the discrepancy.
17///
18/// # Examples
19///
20/// ```ignore
21/// use claude_code_sync::conflict::Conflict;
22/// use claude_code_sync::parser::ConversationSession;
23///
24/// let conflict = Conflict::new(local, remote);
25///
26/// if conflict.is_real_conflict() {
27///     println!("{}", conflict.description());
28/// }
29/// ```
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub struct Conflict {
32    /// The unique identifier for the conversation session that has conflicted.
33    ///
34    /// This ID is shared between both the local and remote versions, as they represent
35    /// different states of the same conversation.
36    pub session_id: String,
37
38    /// The file path to the local version of the conversation.
39    ///
40    /// This points to the conversation file in the local filesystem, typically in the
41    /// user's local conversation storage directory.
42    pub local_file: PathBuf,
43
44    /// The file path to the remote version of the conversation.
45    ///
46    /// This points to the conversation file from the remote source (e.g., synced from
47    /// another machine or cloud storage).
48    pub remote_file: PathBuf,
49
50    /// The timestamp of the most recent message in the local version.
51    ///
52    /// This is `None` if the local conversation has no messages with timestamps.
53    /// The timestamp helps users understand which version is more recent.
54    pub local_timestamp: Option<String>,
55
56    /// The timestamp of the most recent message in the remote version.
57    ///
58    /// This is `None` if the remote conversation has no messages with timestamps.
59    /// The timestamp helps users understand which version is more recent.
60    pub remote_timestamp: Option<String>,
61
62    /// The total number of messages in the local version of the conversation.
63    ///
64    /// This count includes all conversation entries (user messages, assistant responses, etc.)
65    /// and helps users compare the relative completeness of each version.
66    pub local_message_count: usize,
67
68    /// The total number of messages in the remote version of the conversation.
69    ///
70    /// This count includes all conversation entries (user messages, assistant responses, etc.)
71    /// and helps users compare the relative completeness of each version.
72    pub remote_message_count: usize,
73
74    /// A hash of the local conversation's content.
75    ///
76    /// This hash is used to detect whether the local and remote versions are truly different.
77    /// If the hashes match, the conversations are identical despite any metadata differences.
78    pub local_hash: String,
79
80    /// A hash of the remote conversation's content.
81    ///
82    /// This hash is used to detect whether the local and remote versions are truly different.
83    /// If the hashes match, the conversations are identical despite any metadata differences.
84    pub remote_hash: String,
85
86    /// The current resolution status of the conflict.
87    ///
88    /// Initially set to `ConflictResolution::Pending` when a conflict is detected.
89    /// Updated to one of the other variants once the user or system decides how to
90    /// resolve the conflict.
91    pub resolution: ConflictResolution,
92}

94/// Represents the resolution strategy for a conversation conflict.
95///
96/// When a conflict is detected between local and remote versions of the same conversation,
97/// the user or system must choose how to resolve it. This enum captures the different
98/// resolution strategies available.
99///
100/// # Resolution Strategies
101///
102/// - **SmartMerge**: Intelligently combines both versions by merging non-conflicting changes
103/// - **KeepBoth**: Preserves both versions by renaming the remote file to avoid overwriting
104/// - **KeepLocal**: Discards the remote version and keeps only the local version
105/// - **KeepRemote**: Discards the local version and keeps only the remote version
106/// - **Pending**: No resolution has been chosen yet (default state for new conflicts)
107#[derive(Debug, Clone, Serialize, Deserialize)]
108pub enum ConflictResolution {
109    /// Intelligently merge both versions into a single conversation.
110    ///
111    /// This strategy attempts to combine messages from both local and remote versions
112    /// by analyzing message UUIDs, parent relationships, and timestamps. It can handle:
113    /// - Non-overlapping messages (simple merge)
114    /// - Edited messages (resolved by timestamp)
115    /// - Conversation branches (all branches preserved)
116    /// - Entries without UUIDs (merged by timestamp)
117    ///
118    /// If smart merge fails (e.g., due to corrupted data or circular references),
119    /// the system will fall back to another resolution strategy.
120    ///
121    /// # Fields
122    ///
123    /// * `merged_entries` - The result of merging both conversations
124    /// * `stats` - Statistics about the merge operation
125    SmartMerge {
126        /// The merged conversation entries
127        merged_entries: Vec<crate::parser::ConversationEntry>,
128        /// Statistics about the merge operation
129        stats: merge::MergeStats,
130    },
131
132    /// Keep both versions by renaming the remote file with a conflict suffix.
133    ///
134    /// This strategy preserves both the local and remote versions of the conversation.
135    /// The local file retains its original name and location, while the remote file
136    /// is renamed to include a conflict marker (typically a timestamp-based suffix)
137    /// to prevent overwriting the local version.
138    ///
139    /// # Fields
140    ///
141    /// * `renamed_remote_file` - The new path where the remote file will be saved.
142    ///   This path includes a conflict suffix to distinguish it from the local version.
143    ///
144    /// # Example
145    ///
146    /// If the local file is `conversation.jsonl` and a conflict is detected,
147    /// the remote version might be saved as `conversation-conflict-20250122-143000.jsonl`.
148    KeepBoth {
149        /// The destination path for the renamed remote file, including the conflict suffix.
150        renamed_remote_file: PathBuf,
151    },
152
153    /// Keep only the local version and discard the remote version.
154    ///
155    /// This strategy assumes the local version is correct and the remote version
156    /// should be ignored. The local file remains unchanged, and the remote version
157    /// is not saved to disk.
158    KeepLocal,
159
160    /// Keep only the remote version and discard the local version.
161    ///
162    /// This strategy assumes the remote version is correct and should replace the
163    /// local version. The local file will be overwritten with the remote content.
164    KeepRemote,
165
166    /// The conflict has not yet been resolved.
167    ///
168    /// This is the default state for newly detected conflicts. The user must choose
169    /// one of the other resolution strategies before the conflict can be resolved.
170    Pending,
171}

173impl Conflict {
174    /// Creates a new `Conflict` by comparing local and remote conversation sessions.
175    ///
176    /// This function constructs a conflict record from two versions of the same conversation
177    /// session (identified by matching session IDs). It extracts and stores relevant metadata
178    /// from both versions, including timestamps, message counts, and content hashes.
179    ///
180    /// The conflict is initialized with a `Pending` resolution status, indicating that no
181    /// resolution strategy has been chosen yet.
182    ///
183    /// # Arguments
184    ///
185    /// * `local` - A reference to the local version of the conversation session
186    /// * `remote` - A reference to the remote version of the conversation session
187    ///
188    /// # Returns
189    ///
190    /// A new `Conflict` instance containing metadata from both conversation versions,
191    /// with the resolution status set to `Pending`.
192    ///
193    /// # Examples
194    ///
195    /// ```ignore
196    /// use claude_code_sync::conflict::Conflict;
197    /// use claude_code_sync::parser::ConversationSession;
198    ///
199    /// let conflict = Conflict::new(local_session, remote_session);
200    ///
201    /// println!("Conflict in session: {}", conflict.session_id);
202    /// println!("Local messages: {}", conflict.local_message_count);
203    /// println!("Remote messages: {}", conflict.remote_message_count);
204    /// ```
205    pub fn new(local: &ConversationSession, remote: &ConversationSession) -> Self {
206        Conflict {
207            session_id: local.session_id.clone(),
208            local_file: PathBuf::from(&local.file_path),
209            remote_file: PathBuf::from(&remote.file_path),
210            local_timestamp: local.latest_timestamp(),
211            remote_timestamp: remote.latest_timestamp(),
212            local_message_count: local.message_count(),
213            remote_message_count: remote.message_count(),
214            local_hash: local.content_hash(),
215            remote_hash: remote.content_hash(),
216            resolution: ConflictResolution::Pending,
217        }
218    }
219
220    /// Attempts to resolve the conflict using smart merge
221    ///
222    /// This method tries to intelligently combine local and remote versions
223    /// by analyzing message UUIDs, timestamps, and parent relationships.
224    ///
225    /// # Arguments
226    ///
227    /// * `local_session` - The local conversation session
228    /// * `remote_session` - The remote conversation session
229    ///
230    /// # Returns
231    ///
232    /// Returns `Ok(())` if the smart merge succeeds, or an error if it fails.
233    /// On success, the conflict resolution is set to `SmartMerge` with the merged entries.
234    pub fn try_smart_merge(
235        &mut self,
236        local_session: &ConversationSession,
237        remote_session: &ConversationSession,
238    ) -> Result<()> {
239        let merge_result = merge::merge_conversations(local_session, remote_session)?;
240
241        self.resolution = ConflictResolution::SmartMerge {
242            merged_entries: merge_result.merged_entries,
243            stats: merge_result.stats,
244        };
245
246        Ok(())
247    }
248
249    /// Resolve the conflict by keeping both versions
250    pub fn resolve_keep_both(&mut self, conflict_suffix: &str) -> Result<PathBuf> {
251        let remote_file_name = self
252            .remote_file
253            .file_stem()
254            .and_then(|s| s.to_str())
255            .unwrap_or("unknown");
256
257        let remote_file_ext = self
258            .remote_file
259            .extension()
260            .and_then(|s| s.to_str())
261            .unwrap_or("jsonl");
262
263        let parent = self.remote_file.parent().unwrap_or_else(|| Path::new("."));
264
265        let new_name = format!(
266            "{remote_file_name}-{conflict_suffix}.{remote_file_ext}"
267        );
268        let renamed_path = parent.join(new_name);
269
270        self.resolution = ConflictResolution::KeepBoth {
271            renamed_remote_file: renamed_path.clone(),
272        };
273
274        Ok(renamed_path)
275    }
276
277    /// Get a human-readable description of the conflict
278    pub fn description(&self) -> String {
279        format!(
280            "Session {} has diverged:\n  Local: {} messages, last update: {}\n  Remote: {} messages, last update: {}",
281            self.session_id,
282            self.local_message_count,
283            self.local_timestamp.as_deref().unwrap_or("unknown"),
284            self.remote_message_count,
285            self.remote_timestamp.as_deref().unwrap_or("unknown")
286        )
287    }
288
289    /// Determine if this is a real conflict (different content)
290    pub fn is_real_conflict(&self) -> bool {
291        self.local_hash != self.remote_hash
292    }
293}

295/// Conflict detector for conversation sessions
296pub struct ConflictDetector {
297    conflicts: Vec<Conflict>,
298}
299
300impl ConflictDetector {
301    /// Creates a new `ConflictDetector` with an empty conflict list.
302    ///
303    /// The conflict detector is used to identify and manage conflicts between local and remote
304    /// conversation sessions. It maintains a list of detected conflicts and provides methods
305    /// to resolve them according to different strategies.
306    ///
307    /// The detector starts with no conflicts; conflicts are added by calling the [`detect`]
308    /// method with local and remote conversation sessions.
309    ///
310    /// [`detect`]: ConflictDetector::detect
311    ///
312    /// # Returns
313    ///
314    /// A new `ConflictDetector` instance with an empty internal conflict list, ready to
315    /// detect and manage conflicts between conversation sessions.
316    ///
317    /// # Examples
318    ///
319    /// ```
320    /// # use claude_code_sync::conflict::ConflictDetector;
321    /// # use claude_code_sync::parser::ConversationSession;
322    /// # fn example(local_sessions: Vec<ConversationSession>, remote_sessions: Vec<ConversationSession>) {
323    /// let mut detector = ConflictDetector::new();
324    ///
325    /// // Detect conflicts between local and remote sessions
326    /// detector.detect(&local_sessions, &remote_sessions);
327    ///
328    /// if detector.has_conflicts() {
329    ///     println!("Found {} conflicts", detector.conflict_count());
330    /// }
331    /// # }
332    /// ```
333    ///
334    /// # See Also
335    ///
336    /// * [`detect`] - Method to scan for conflicts between local and remote sessions
337    /// * [`has_conflicts`] - Check if any conflicts have been detected
338    /// * [`conflict_count`] - Get the number of detected conflicts
339    ///
340    /// [`detect`]: ConflictDetector::detect
341    /// [`has_conflicts`]: ConflictDetector::has_conflicts
342    /// [`conflict_count`]: ConflictDetector::conflict_count
343    pub fn new() -> Self {
344        ConflictDetector {
345            conflicts: Vec::new(),
346        }
347    }
348
349    /// Compare local and remote sessions and detect conflicts
350    pub fn detect(
351        &mut self,
352        local_sessions: &[ConversationSession],
353        remote_sessions: &[ConversationSession],
354    ) {
355        // Build a map of session_id -> local session
356        let local_map: std::collections::HashMap<_, _> = local_sessions
357            .iter()
358            .map(|s| (s.session_id.clone(), s))
359            .collect();
360
361        // Check each remote session against local
362        for remote in remote_sessions {
363            if let Some(local) = local_map.get(&remote.session_id) {
364                // Session exists in both - check for conflicts
365                if local.content_hash() != remote.content_hash() {
366                    let conflict = Conflict::new(local, remote);
367                    if conflict.is_real_conflict() {
368                        self.conflicts.push(conflict);
369                    }
370                }
371            }
372        }
373    }
374
375    /// Resolve all conflicts using the "keep both" strategy
376    #[allow(dead_code)]
377    pub fn resolve_all_keep_both(&mut self) -> Result<Vec<(PathBuf, PathBuf)>> {
378        let mut renames = Vec::new();
379
380        for conflict in &mut self.conflicts {
381            let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
382            let conflict_suffix = format!("conflict-{timestamp}");
383
384            let renamed_path = conflict.resolve_keep_both(&conflict_suffix)?;
385            renames.push((conflict.remote_file.clone(), renamed_path));
386        }
387
388        Ok(renames)
389    }
390
391    /// Get all detected conflicts
392    pub fn conflicts(&self) -> &[Conflict] {
393        &self.conflicts
394    }
395
396    /// Get mutable reference to all detected conflicts
397    pub fn conflicts_mut(&mut self) -> &mut [Conflict] {
398        &mut self.conflicts
399    }
400
401    /// Check if any conflicts were detected
402    pub fn has_conflicts(&self) -> bool {
403        !self.conflicts.is_empty()
404    }
405
406    /// Get count of conflicts
407    pub fn conflict_count(&self) -> usize {
408        self.conflicts.len()
409    }
410}

423    fn create_test_session(session_id: &str, message_count: usize) -> ConversationSession {
424        let mut entries = Vec::new();
425
426        for i in 0..message_count {
427            entries.push(ConversationEntry {
428                entry_type: if i % 2 == 0 { "user" } else { "assistant" }.to_string(),
429                uuid: Some(format!("uuid-{i}")),
430                parent_uuid: if i > 0 {
431                    Some(format!("uuid-{}", i - 1))
432                } else {
433                    None
434                },
435                session_id: Some(session_id.to_string()),
436                timestamp: Some(format!("2025-01-01T{i:02}:00:00Z")),
437                message: None,
438                cwd: None,
439                version: None,
440                git_branch: None,
441                extra: serde_json::Value::Null,
442            });
443        }
444
445        ConversationSession {
446            session_id: session_id.to_string(),
447            entries,
448            file_path: format!("/test/{session_id}.jsonl"),
449        }
450    }

452    #[test]
453    fn test_conflict_detection() {
454        let local_session = create_test_session("session-1", 5);
455        let remote_session = create_test_session("session-1", 6);
456
457        let mut detector = ConflictDetector::new();
458        detector.detect(&[local_session], &[remote_session]);

466        assert_eq!(conflict.remote_message_count, 6);
