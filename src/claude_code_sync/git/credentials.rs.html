1use git2::{Cred, CredentialType};
2
3/// Credential callback for git operations
4///
5/// This function is used by git2 to request credentials during remote operations.
6/// It tries multiple authentication methods in order:
7/// 1. Git credential helper (reads from git config)
8/// 2. SSH agent (for SSH URLs)
9pub(super) fn credential_callback(
10    _url: &str,
11    username_from_url: Option<&str>,
12    _allowed_types: CredentialType,
13) -> Result<Cred, git2::Error> {
14    // Try credential helper first
15    git2::Cred::credential_helper(&git2::Config::open_default()?, _url, username_from_url)
16        // Fall back to SSH agent if credential helper fails
17        .or_else(|_| git2::Cred::ssh_key_from_agent(username_from_url.unwrap_or("git")))
18}
