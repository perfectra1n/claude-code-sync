1//! The category registry: one row per syncable artifact kind. Everything the
2//! engine does — collecting sources, choosing repo destinations, picking a
3//! merge strategy — is driven by this data.
4
5use serde::{Deserialize, Serialize};
6
7/// Identifies one syncable artifact category.
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
9#[serde(rename_all = "kebab-case")]
10pub enum CategoryId {
11    Settings,
12    Memory,
13    Skills,
14    Agents,
15    Commands,
16    Plugins,
17    Plans,
18    Todos,
19    PromptHistory,
20    /// Non-JSONL files inside `~/.claude/projects/` (images, PDFs, per-project
21    /// memory). Gated by `!exclude_attachments` instead of a toggle, honoring
22    /// the long-documented attachments contract.
23    ProjectAttachments,
24}
25
26/// What to copy, relative to `~/.claude`. Missing sources are treated as
27/// empty, never as an error — a machine may simply not have e.g. `todos/`.
28#[derive(Debug, Clone, Copy, PartialEq, Eq)]
29pub enum SourceSpec {
30    /// Exact files. Anything not listed is structurally unreachable — this is
31    /// how `plugins/` syncs its two manifests but never `cache/`.
32    Files(&'static [&'static str]),
33    /// A directory walked recursively (symlinks not followed).
34    Dir(&'static str),
35}
36
37/// How pull reconciles a repo copy with the local file.
38#[derive(Debug, Clone, Copy, PartialEq, Eq)]
39pub enum MergeStrategy {
40    /// Remote bytes win (only written when they differ; snapshot first).
41    RawOverwrite,
42    /// Line-wise union of both sides (append-only JSONL like history.jsonl);
43    /// applied on push and pull so machines converge grow-only.
44    UnionJsonl,
45}
46
47/// Where a category's files live inside the sync repository.
48#[derive(Debug, Clone, Copy, PartialEq, Eq)]
49pub enum DestRoot {
50    /// `<repo>/artifacts/<repo_subdir>/...` — the default for all categories.
51    Artifacts,
52    /// `<repo>/<sync_subdirectory>/...` — shares the session tree (used only
53    /// by project attachments, which live beside their transcripts).
54    SessionTree,
55}
56
57/// One registry row.
58#[derive(Debug)]
59pub struct CategoryDescriptor {
60    pub id: CategoryId,
61    /// Config key / CLI name (kebab-case).
62    pub name: &'static str,
63    /// Subdirectory under the sync repo's `artifacts/` root (unused for
64    /// `DestRoot::SessionTree`).
65    pub repo_subdir: &'static str,
66    pub source: SourceSpec,
67    pub merge: MergeStrategy,
68    pub dest: DestRoot,
69    /// File extensions this category never copies (e.g. attachments skip
70    /// `.jsonl`, which belongs to the session pipeline).
71    pub exclude_extensions: &'static [&'static str],
72    /// Short human description for wizard / config --show.
73    pub description: &'static str,
74}
75
76/// Root directory inside the sync repository for all artifact categories.
77/// Deliberately a constant: a fixed category→path mapping keeps pull
78/// deterministic and the deny-guarantee auditable.
79pub const ARTIFACTS_SUBDIR: &str = "artifacts";

187/// Per-category enable switches, stored in the `[sync_artifacts]` table of
188/// config.toml. Every field defaults to false so configs written by older
189/// versions parse unchanged and existing users opt in explicitly.
190#[derive(Debug, Clone, Default, Serialize, Deserialize, PartialEq, Eq)]
191pub struct ArtifactToggles {
192    #[serde(default)]
193    pub settings: bool,
194    #[serde(default)]
195    pub memory: bool,
196    #[serde(default)]
197    pub skills: bool,
198    #[serde(default)]
199    pub agents: bool,
200    #[serde(default)]
201    pub commands: bool,
202    #[serde(default)]
203    pub plugins: bool,
204    #[serde(default)]
205    pub plans: bool,
206    #[serde(default)]
207    pub todos: bool,
208    #[serde(default)]
209    pub prompt_history: bool,
210}
211
212impl ArtifactToggles {
213    /// A toggles value with every category enabled (used by onboarding
214    /// defaults and the `all` CLI shorthand).
215    pub fn all_enabled() -> Self {
216        Self {
217            settings: true,
218            memory: true,
219            skills: true,
220            agents: true,
221            commands: true,
222            plugins: true,
223            plans: true,
224            todos: true,
225            prompt_history: true,
226        }
227    }
228
229    /// Whether any category is enabled at all.
230    pub fn any_enabled(&self) -> bool {
231        REGISTRY.iter().any(|d| self.is_enabled(d.id))
232    }
233
234    /// Read the switch for one category.
235    pub fn is_enabled(&self, id: CategoryId) -> bool {
236        match id {
237            CategoryId::Settings => self.settings,
238            CategoryId::Memory => self.memory,
239            CategoryId::Skills => self.skills,
240            CategoryId::Agents => self.agents,
241            CategoryId::Commands => self.commands,
242            CategoryId::Plugins => self.plugins,
243            CategoryId::Plans => self.plans,
244            CategoryId::Todos => self.todos,
245            CategoryId::PromptHistory => self.prompt_history,
246            // Attachments are gated by FilterConfig::exclude_attachments, not
247            // by a toggle; as a toggle they always read as off.
248            CategoryId::ProjectAttachments => false,
249        }
250    }
251
252    /// Flip the switch for one category.
253    pub fn set_enabled(&mut self, id: CategoryId, value: bool) {
254        match id {
255            CategoryId::Settings => self.settings = value,
256            CategoryId::Memory => self.memory = value,
257            CategoryId::Skills => self.skills = value,
258            CategoryId::Agents => self.agents = value,
259            CategoryId::Commands => self.commands = value,
260            CategoryId::Plugins => self.plugins = value,
261            CategoryId::Plans => self.plans = value,
262            CategoryId::Todos => self.todos = value,
263            CategoryId::PromptHistory => self.prompt_history = value,
264            CategoryId::ProjectAttachments => {}
265        }
266    }
267}

269/// Look up a descriptor by its CLI/config name (kebab-case).
270pub fn find_by_name(name: &str) -> Option<&'static CategoryDescriptor> {
271    REGISTRY.iter().find(|d| d.name == name)
272}
273
274/// Registry rows driven by ArtifactToggles — everything except attachments,
275/// which the long-standing exclude_attachments flag governs.
276pub fn toggleable() -> impl Iterator<Item = &'static CategoryDescriptor> {
277    REGISTRY
278        .iter()
279        .filter(|d| d.id != CategoryId::ProjectAttachments)
280}
281
282/// All registry rows whose toggle is on.
283#[allow(dead_code)] // used via the library target; the bin compiles this module separately
284pub fn enabled_categories(toggles: &ArtifactToggles) -> Vec<&'static CategoryDescriptor> {
285    REGISTRY
286        .iter()
287        .filter(|d| toggles.is_enabled(d.id))
288        .collect()
289}

297    fn test_registry_has_all_ten_categories() {

306        let subdirs: HashSet<_> = REGISTRY.iter().map(|d| d.repo_subdir).collect();
307        assert_eq!(names.len(), REGISTRY.len());

311    #[test]
312    fn test_plugins_category_is_exact_file_allowlist() {
313        let plugins = REGISTRY
314            .iter()
315            .find(|d| d.id == CategoryId::Plugins)
316            .unwrap();
317        match plugins.source {

321                    [
322                        "plugins/installed_plugins.json",
323                        "plugins/known_marketplaces.json"

326            }
327            _ => panic!("plugins must be an exact file allowlist, never a dir walk"),
328        }

331    #[test]
332    fn test_prompt_history_uses_union_merge() {
333        let ph = REGISTRY
334            .iter()
335            .find(|d| d.id == CategoryId::PromptHistory)
336            .unwrap();
337        assert_eq!(ph.merge, MergeStrategy::UnionJsonl);
338        assert_eq!(ph.source, SourceSpec::Files(&["history.jsonl"]));
339        // Everything else raw-overwrites
340        for d in REGISTRY
341            .iter()
342            .filter(|d| d.id != CategoryId::PromptHistory)
343        {
344            assert_eq!(d.merge, MergeStrategy::RawOverwrite, "{}", d.name);
345        }
346    }

348    #[test]
349    fn test_settings_never_includes_local_overrides() {
350        let settings = REGISTRY
351            .iter()
352            .find(|d| d.id == CategoryId::Settings)
353            .unwrap();
354        match settings.source {
355            SourceSpec::Files(files) => {
356                assert!(files.contains(&"settings.json"));
357                assert!(files.contains(&"keybindings.json"));
358                assert!(!files.contains(&"settings.local.json"));
359            }
360            _ => panic!("settings must be an exact file allowlist"),
361        }

363
364    #[test]
365    fn test_toggles_map_every_category() {
366        let mut toggles = ArtifactToggles::default();
367        assert!(!toggles.any_enabled());
368        for d in toggleable() {
369            assert!(!toggles.is_enabled(d.id), "{} defaults off", d.name);
370            toggles.set_enabled(d.id, true);
371            assert!(toggles.is_enabled(d.id), "{} can be enabled", d.name);
372        }
373        assert_eq!(toggles, ArtifactToggles::all_enabled());
374        // Attachments are flag-governed, not toggleable.
375        assert!(!toggles.is_enabled(CategoryId::ProjectAttachments));
376        toggles.set_enabled(CategoryId::ProjectAttachments, true);
377        assert!(!toggles.is_enabled(CategoryId::ProjectAttachments));
378    }

385        }

394        let toggles: ArtifactToggles = toml::from_str("").unwrap();
