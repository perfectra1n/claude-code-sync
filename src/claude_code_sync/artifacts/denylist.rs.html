1//! Hardcoded never-sync rules for artifact copies.
2//!
3//! These are enforced at the single copy chokepoint in both directions: on
4//! push nothing denied leaves the machine, and on pull a poisoned sync repo
5//! cannot plant a denied file (or escape `~/.claude`). No user configuration
6//! is consulted here — there is deliberately no way to override this list.
7
8use std::path::Path;
9
10/// Exact file names that must never sync, regardless of location.
11const DENIED_FILE_NAMES: &[&str] = &[
12    ".credentials.json",
13    "settings.local.json",
14    ".claude.json",
15    "stats-cache.json",
16    ".last-update-result.json",
17    "mcp-needs-auth-cache.json",
18];
19
20/// File extensions that must never sync (key material).
21const DENIED_EXTENSIONS: &[&str] = &["pem", "key"];
22
23/// Name prefixes that must never sync (`.env*`, `daemon`, `daemon.lock`, ...).
24const DENIED_NAME_PREFIXES: &[&str] = &[".env", "daemon"];
25
26/// Directory names that are machine-local or cache-like; any path containing
27/// one of these components is denied.
28const DENIED_DIR_COMPONENTS: &[&str] = &[
29    "shell-snapshots",
30    "session-env",
31    "file-history",
32    "paste-cache",
33    "cache",
34    "debug",
35    "statsig",
36    "backups",
37    "sessions",
38];
39
40/// Returns true when a `~/.claude`-relative path must never be copied by
41/// artifact sync, checking every path component against the deny rules.
42pub fn is_denied(rel_path: &Path) -> bool {
43    for component in rel_path.components() {
44        let Some(name) = component.as_os_str().to_str() else {
45            // Non-UTF-8 component: refuse rather than guess.
46            return true;
47        };
48
49        if DENIED_FILE_NAMES.contains(&name) || DENIED_DIR_COMPONENTS.contains(&name) {
50            return true;
51        }
52
53        if DENIED_NAME_PREFIXES
54            .iter()
55            .any(|prefix| name.starts_with(prefix))
56        {
57            return true;
58        }
59
60        if let Some(ext) = Path::new(name).extension().and_then(|e| e.to_str()) {
61            if DENIED_EXTENSIONS
62                .iter()
63                .any(|denied| ext.eq_ignore_ascii_case(denied))
64            {
65                return true;
66            }
67        }
68    }
69
70    false
71}
72
73/// Returns true when a repo-relative path is unsafe to restore: absolute,
74/// or containing `..`/prefix components that could escape the target root.
75pub fn is_unsafe_rel_path(rel_path: &Path) -> bool {
76    use std::path::Component;
77
78    rel_path.components().any(|c| {
79        matches!(
80            c,
81            Component::ParentDir | Component::RootDir | Component::Prefix(_)
82        )
83    })
84}

91    #[test]
92    fn test_denies_credential_files_anywhere() {
93        assert!(is_denied(Path::new(".credentials.json")));
94        assert!(is_denied(Path::new("skills/x/.credentials.json")));
95        assert!(is_denied(Path::new("settings.local.json")));
96        assert!(is_denied(Path::new("plans/settings.local.json")));
97        assert!(is_denied(Path::new(".claude.json")));
98    }

104        assert!(is_denied(Path::new("agents/x/cert.PEM"))); // case-insensitive
105    }
