1//! Git LFS support for large conversation files.
2//!
3//! This module provides helpers for configuring Git LFS. Since we use
4//! the git CLI, LFS operations (clean/smudge) happen automatically
5//! once configured. This module handles the setup.
6
7use anyhow::{bail, Context, Result};
8use std::fs;
9use std::path::Path;
10use std::process::Command;
11
12/// Check if git-lfs is installed on the system.
13pub fn is_installed() -> bool {
14    Command::new("git-lfs")
15        .arg("version")
16        .output()
17        .map(|o| o.status.success())
18        .unwrap_or(false)
19}
20
21/// Initialize LFS in a repository.
22///
23/// This runs `git lfs install --local` to configure the repository for LFS.
24pub fn init(repo_path: &Path) -> Result<()> {
25    let output = Command::new("git")
26        .args(["lfs", "install", "--local"])
27        .current_dir(repo_path)
28        .output()
29        .context("Failed to run 'git lfs install'")?;
30
31    if !output.status.success() {
32        bail!(
33            "git lfs install failed: {}",
34            String::from_utf8_lossy(&output.stderr)
35        );
36    }
37
38    Ok(())
39}

41/// Configure .gitattributes for LFS patterns.
42///
43/// Writes a .gitattributes file that tells git to use LFS for the
44/// specified file patterns.
45pub fn configure_gitattributes(repo_path: &Path, patterns: &[String]) -> Result<()> {
46    let gitattributes_path = repo_path.join(".gitattributes");
47
48    let mut content = String::new();
49
50    // Read existing content if file exists
51    if gitattributes_path.exists() {
52        content = fs::read_to_string(&gitattributes_path)
53            .context("Failed to read existing .gitattributes")?;
54    }
55
56    // Add LFS patterns that aren't already present
57    for pattern in patterns {
58        let lfs_line = format!("{} filter=lfs diff=lfs merge=lfs -text", pattern);
59        if !content.contains(&lfs_line) {
60            if !content.is_empty() && !content.ends_with('\n') {
61                content.push('\n');
62            }
63            content.push_str(&lfs_line);
64            content.push('\n');
65        }
66    }
67
68    fs::write(&gitattributes_path, content).context("Failed to write .gitattributes")?;
69
70    Ok(())
71}

73/// Set up LFS for a repository with the given patterns.
74///
75/// This is a convenience function that:
76/// 1. Checks if git-lfs is installed
77/// 2. Initializes LFS in the repository
78/// 3. Configures .gitattributes for the patterns
79pub fn setup(repo_path: &Path, patterns: &[String]) -> Result<()> {
80    if !is_installed() {
81        bail!(
82            "git-lfs is not installed.\n\
83            Install it with:\n  \
84            - macOS: brew install git-lfs\n  \
85            - Ubuntu/Debian: apt install git-lfs\n  \
86            - Windows: https://git-lfs.github.com"
87        );
88    }
89
90    init(repo_path)?;
91    configure_gitattributes(repo_path, patterns)?;
92
93    Ok(())
94}

96#[cfg(test)]
97mod tests {

99    use tempfile::TempDir;

101    #[test]
102    fn test_configure_gitattributes_new_file() {
103        let temp_dir = TempDir::new().unwrap();
104        let patterns = vec!["*.jsonl".to_string()];
105
106        configure_gitattributes(temp_dir.path(), &patterns).unwrap();
107
108        let content = fs::read_to_string(temp_dir.path().join(".gitattributes")).unwrap();
109        assert!(content.contains("*.jsonl filter=lfs diff=lfs merge=lfs -text"));
110    }
111
112    #[test]
113    fn test_configure_gitattributes_existing_file() {
114        let temp_dir = TempDir::new().unwrap();
115        let gitattributes = temp_dir.path().join(".gitattributes");
116
117        // Write existing content
118        fs::write(&gitattributes, "*.txt text\n").unwrap();
119
120        let patterns = vec!["*.jsonl".to_string()];
121        configure_gitattributes(temp_dir.path(), &patterns).unwrap();
122
123        let content = fs::read_to_string(&gitattributes).unwrap();
124        assert!(content.contains("*.txt text"));
125        assert!(content.contains("*.jsonl filter=lfs diff=lfs merge=lfs -text"));
126    }

148    #[test]
149    fn test_setup_configures_gitattributes() {
150        let temp_dir = TempDir::new().unwrap();
151        let patterns = vec!["*.jsonl".to_string(), "*.png".to_string()];
152
153        // Initialize a git repo first (setup requires a git repo)
154        std::process::Command::new("git")
155            .args(["init"])
156            .current_dir(temp_dir.path())
157            .output()
158            .unwrap();
159
160        // Setup may fail if git-lfs is not installed, which is OK for this test
161        // We just want to verify it configures .gitattributes when it can
162        let result = setup(temp_dir.path(), &patterns);
163
164        if is_installed() {
165            // If LFS is installed, setup should succeed
166            assert!(result.is_ok(), "setup failed: {:?}", result.err());
167
168            // Verify .gitattributes was configured
169            let content = fs::read_to_string(temp_dir.path().join(".gitattributes")).unwrap();
170            assert!(content.contains("*.jsonl filter=lfs"));
171            assert!(content.contains("*.png filter=lfs"));
172        } else {
173            // If LFS is not installed, setup should fail with helpful message
174            assert!(result.is_err());
175            let err = result.unwrap_err().to_string();
176            assert!(err.contains("git-lfs is not installed"));
177        }
178    }
